27/// Camera update routine.
28pub fn update_camera(
29    window_q: Query<&Window>,
30    mut ev_motion: EventReader<MouseMotion>,
31    mut ev_scroll: EventReader<MouseWheel>,
32    input_mouse: Res<ButtonInput<MouseButton>>,
33    mut query: Query<(&mut PanOrbitCamera, &mut Transform, &Projection)>,
34) {
35    let window = window_q.single();
36    // change input mapping for orbit and panning here
37    let orbit_button = MouseButton::Right;
38    let pan_button = MouseButton::Left;
39
40    let mut pan = Vec2::ZERO;
41    let mut rotation_move = Vec2::ZERO;
42    let mut scroll = 0.0;
43    let mut orbit_button_changed = false;
44
45    if input_mouse.pressed(orbit_button) {
46        for ev in ev_motion.read() {
47            rotation_move += ev.delta;
48        }
49    } else if input_mouse.pressed(pan_button) {
50        // Pan only if we're not rotating at the moment
51        for ev in ev_motion.read() {
52            pan += ev.delta * 2.;
53        }
54    }
55
56    for ev in ev_scroll.read() {
57        scroll += ev.y;
58
59        scroll /= if cfg!(target_arch = "wasm32") {
60            100.0
61        } else {
62            20.0
63        };
64    }
65    if input_mouse.just_released(orbit_button) || input_mouse.just_pressed(orbit_button) {
66        orbit_button_changed = true;
67    }
68
69    for (mut pan_orbit, mut transform, projection) in &mut query {
70        if orbit_button_changed {
71            // only check for upside down when orbiting started or ended this frame
72            // if the camera is "upside" down, panning horizontally would be inverted, so invert the input to make it correct
73            let up = transform.rotation * Vec3::Y;
74            pan_orbit.upside_down = up.y <= 0.0;
75        }
76        let window = vec2(
77            window.physical_width() as f32,
78            window.physical_height() as f32,
79        );
80
81        let mut any = false;
82        if rotation_move.length_squared() > 0.0 {
83            any = true;
84            let delta_x = {
85                let delta = rotation_move.x / window.x * std::f32::consts::PI * 2.0;
86                if pan_orbit.upside_down { -delta } else { delta }
87            };
88            let delta_y = rotation_move.y / window.y * std::f32::consts::PI;
89            let yaw = Quat::from_rotation_y(-delta_x);
90            let pitch = Quat::from_rotation_x(-delta_y);
91            transform.rotation = yaw * transform.rotation; // rotate around global y axis
92            transform.rotation *= pitch; // rotate around local x axis
93        } else if pan.length_squared() > 0.0 {
94            any = true;
95
96            if let Projection::Perspective(projection) = projection {
97                pan *= Vec2::new(projection.fov * projection.aspect_ratio, projection.fov) / window;
98            }
99            // translate by local axes
100            let right = transform.rotation * Vec3::X * -pan.x;
101            let up = transform.rotation * Vec3::Y * pan.y;
102            // make panning proportional to distance away from focus point
103            let translation = (right + up) * pan_orbit.radius;
104            pan_orbit.focus += translation;
105        } else if scroll.abs() > 0.0 {
106            any = true;
107            pan_orbit.radius -= scroll * pan_orbit.radius * 0.2;
108            // dont allow zoom to reach zero or you get stuck
109            pan_orbit.radius = f32::max(pan_orbit.radius, 0.05);
110        }
111
112        if any {
113            let rot_matrix = Mat3::from_quat(transform.rotation);
114            transform.translation =
115                pan_orbit.focus + rot_matrix.mul_vec3(Vec3::new(0.0, 0.0, pan_orbit.radius));
116        }
117    }
118
119    ev_motion.clear();
120}

123/// Detects if the cursor is positioned in the render tab.
124///
125/// This is used to ignore camera related input when interacting with something other than the
126pub fn cursor_in_render(
127    q_windows: Query<&Window, With<PrimaryWindow>>,
128    ui_state: Res<UiState>,
129) -> bool {
130    // returns true if and only if the cursor is positioned in the render tab
131    if let Some(position) = q_windows.single().cursor_position() {
132        let tree = ui_state.tree();
133        if let Some((node_idx, _)) = tree.find_tab(&CustomTab::Render) {
134            let rect = &tree[node_idx].rect().expect("unreachable");
135            return rect.contains(position.to_array().into());
136        }
137        return false;
138    }
139    false
140}
