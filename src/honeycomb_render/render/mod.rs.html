18/// Plugin handling scene setup and updates.
19pub struct ScenePlugin;
20
21impl Plugin for ScenePlugin {
22    fn build(&self, app: &mut App) {
23        // camera
24        app.add_systems(Startup, scene::setup_scene).add_systems(
25            Update,
26            camera::update_camera.run_if(camera::cursor_in_render),
27        );
28
29        // picking
30        app.add_plugins(DefaultPickingPlugins.build())
31            .add_plugins(OutlinePlugin)
32            .insert_resource(SelectionPluginSettings::default())
33            .add_systems(Update, picking::update_picking);
34
35        // update displayed content
36
37        // FocusedCapture change
38        app.add_systems(
39            Update,
40            (
41                update::dart_render,
42                update::vertices_render,
43                update::edges_render,
44            )
45                .run_if(
46                    resource_changed::<FocusedCapture>
47                        .and_then(not(resource_added::<FocusedCapture>))
48                        .and_then(input_just_released(MouseButton::Left)),
49                ),
50        );
51
52        // dart updates
53        app.add_systems(
54            Update,
55            (update::dart_render, update::dart_mat_handle).run_if(
56                resource_changed::<DartRenderColor>
57                    .and_then(not(resource_added::<DartRenderColor>)),
58            ),
59        );
60        app.add_systems(
61            Update,
62            update::dart_heads_handle.run_if(
63                resource_changed::<DartWidth>
64                    .and_then(not(resource_added::<DartWidth>))
65                    .and_then(input_just_released(MouseButton::Left))
66                    .or_else(
67                        resource_changed::<DartHeadMul>
68                            .and_then(not(resource_added::<DartHeadMul>))
69                            .and_then(input_just_released(MouseButton::Left)),
70                    ),
71            ),
72        );
73        app.add_systems(
74            Update,
75            update::dart_bodies_handles.run_if(
76                resource_changed::<DartWidth>
77                    .and_then(not(resource_added::<DartWidth>))
78                    .and_then(input_just_released(MouseButton::Left))
79                    .or_else(
80                        resource_changed::<DartShrink>
81                            .and_then(not(resource_added::<DartShrink>))
82                            .and_then(input_just_released(MouseButton::Left)),
83                    ),
84            ),
85        );
86        app.add_systems(
87            Update,
88            (update::dart_heads_transform, update::dart_bodies_transform).run_if(
89                resource_changed::<DartShrink>
90                    .and_then(not(resource_added::<DartShrink>))
91                    .and_then(input_just_released(MouseButton::Left)),
92            ),
93        );
94
95        // vertex updates
96        app.add_systems(
97            Update,
98            (update::vertices_render, update::vertices_mat_handle).run_if(
99                resource_changed::<VertexRenderColor>
100                    .and_then(not(resource_added::<VertexRenderColor>)),
101            ),
102        );
103        app.add_systems(
104            Update,
105            update::vertices_handle.run_if(
106                resource_changed::<VertexWidth>
107                    .and_then(not(resource_added::<VertexWidth>))
108                    .and_then(input_just_released(MouseButton::Left)),
109            ),
110        );
111        // edge updates
112        app.add_systems(
113            Update,
114            (update::edges_render, update::edges_mat_handle).run_if(
115                resource_changed::<EdgeRenderColor>
116                    .and_then(not(resource_added::<EdgeRenderColor>)),
117            ),
118        );
119        app.add_systems(
120            Update,
121            update::edges_handle.run_if(
122                resource_changed::<EdgeWidth>
123                    .and_then(not(resource_added::<EdgeWidth>))
124                    .and_then(input_just_released(MouseButton::Left)),
125            ),
126        );
127    }
128}
