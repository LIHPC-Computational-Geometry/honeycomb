1//! # Description
2//!
3//! ## Routine
4//!
5//! The algorithm fetches all vertices that are not on the border of the map, fetch all identifiers
6//! of each respective vertices' neighbors. Then, for all vertices:
7//!
8//! - compute the average between neighbors
9//! - overwrite current vertex value with computed average
10//!
11//! ## Benchmark
12//!
13//! This binary is meant to be use to evaluate scalability of geometry-only kernels. It is
14//! parallelized using rayon, and the number of thread used for execution can be controlled using
15//! `taskset`. By controlling this, and the grid size, we can evaluate both strong and weak
16//! scaling characteristics.
17
18use rayon::prelude::*;
19
20use honeycomb::core::stm::atomically;
21use honeycomb::prelude::{
22    CMap2, CMapBuilder, CoordsFloat, DartIdType, Orbit2, OrbitPolicy, Vertex2, VertexIdType,
23    NULL_DART_ID,
24};
25
26use crate::cli::ShiftArgs;
27
28pub fn bench_shift<T: CoordsFloat>(args: ShiftArgs) -> CMap2<T> {
29    let map: CMap2<T> = CMapBuilder::from(args.input).build().unwrap();
30
31    if args.no_conflict {
32        todo!("TODO: require a partitioning algorithm")
33    } else {
34        // fetch all vertices that are not on the boundary of the map
35        let tmp: Vec<(VertexIdType, Vec<VertexIdType>)> = map
36            .iter_vertices()
37            .filter_map(|v| {
38                if Orbit2::new(&map, OrbitPolicy::Vertex, v as DartIdType)
39                    .any(|d| map.beta::<2>(d) == NULL_DART_ID)
40                {
41                    None
42                } else {
43                    Some((
44                        v,
45                        Orbit2::new(&map, OrbitPolicy::Vertex, v as DartIdType)
46                            .map(|d| map.vertex_id(map.beta::<2>(d)))
47                            .collect(),
48                    ))
49                }
50            })
51            .collect();
52        // main loop
53        let mut round = 0;
54        loop {
55            tmp.par_iter().for_each(|(vid, neigh)| {
56                atomically(|trans| {
57                    let mut new_val = Vertex2::default();
58                    for v in neigh {
59                        let vertex = map.read_vertex(trans, *v)?.unwrap();
60                        new_val.0 += vertex.0;
61                        new_val.1 += vertex.1;
62                    }
63                    new_val.0 /= T::from(neigh.len()).unwrap();
64                    new_val.1 /= T::from(neigh.len()).unwrap();
65                    map.write_vertex(trans, *vid, new_val)
66                });
67            });
68
69            round += 1;
70            if round >= args.n_rounds.get() {
71                break;
72            }
73        }
74    }
75
76    map
77}
