7/// Internal grid-building routine
8#[allow(clippy::cast_possible_truncation, clippy::too_many_lines)]
9pub(crate) fn build_2d_grid<T: CoordsFloat>(
10    builder: CMapBuilder<2>,
11    origin: Vertex2<T>,
12    [n_square_x, n_square_y]: [usize; 2],
13    [len_per_x, len_per_y]: [T; 2],
14) -> CMap2<T> {
15    let map: CMap2<T> =
16        CMapBuilder::from_n_darts_and_attributes(4 * n_square_x * n_square_y, builder)
17            .build()
18            .unwrap();
19
20    // init beta functions
21    (1..=(4 * n_square_x * n_square_y) as DartIdType)
22        .zip(generate_square_beta_values(n_square_x, n_square_y))
23        .par_bridge()
24        .for_each(|(dart, images)| {
25            map.set_betas(dart, images);
26        });
27
28    // place vertices
29
30    // bottow left vertex of all cells
31    (0..n_square_y)
32        // flatten the loop to expose more parallelism
33        .flat_map(|y_idx| (0..n_square_x).map(move |x_idx| (y_idx, x_idx)))
34        .par_bridge()
35        .for_each(|(y_idx, x_idx)| {
36            let vertex_id = map.vertex_id((1 + x_idx * 4 + y_idx * 4 * n_square_x) as DartIdType);
37            map.force_write_vertex(
38                vertex_id,
39                origin
40                    + Vector2(
41                        T::from(x_idx).unwrap() * len_per_x,
42                        T::from(y_idx).unwrap() * len_per_y,
43                    ),
44            );
45        });
46
47    // top left vertex of all top row cells
48    (0..n_square_x).into_par_iter().for_each(|x_idx| {
49        let y_idx = n_square_y - 1;
50        let vertex_id = map.vertex_id((4 + x_idx * 4 + y_idx * 4 * n_square_x) as DartIdType);
51        map.force_write_vertex(
52            vertex_id,
53            origin
54                + Vector2(
55                    T::from(x_idx).unwrap() * len_per_x,
56                    T::from(y_idx + 1).unwrap() * len_per_y,
57                ),
58        );
59    });
60
61    // bottom right vertex of all right col cells
62    (0..n_square_y).into_par_iter().for_each(|y_idx| {
63        let x_idx = n_square_x - 1;
64        let vertex_id = map.vertex_id((2 + x_idx * 4 + y_idx * 4 * n_square_x) as DartIdType);
65        map.force_write_vertex(
66            vertex_id,
67            origin
68                + Vector2(
69                    T::from(x_idx + 1).unwrap() * len_per_x,
70                    T::from(y_idx).unwrap() * len_per_y,
71                ),
72        );
73    });
74
75    // top right vertex of the last cell
76    {
77        let (x_idx, y_idx) = (n_square_x - 1, n_square_y - 1);
78        let vertex_id = map.vertex_id((3 + x_idx * 4 + y_idx * 4 * n_square_x) as DartIdType); // top right
79        map.force_write_vertex(
80            vertex_id,
81            origin
82                + Vector2(
83                    T::from(x_idx + 1).unwrap() * len_per_x,
84                    T::from(y_idx + 1).unwrap() * len_per_y,
85                ),
86        );
87    }
88
89    // check the number of built faces
90    // this is set as debug only because the operation cost scales with map size
91    // this can quickly overshadow the exectime of all previous code
92    debug_assert_eq!(map.iter_faces().count(), n_square_x * n_square_y);
93
94    map
95}

97#[allow(clippy::cast_possible_truncation,clippy::inline_always)]
98#[rustfmt::skip]
99#[inline(always)]
100fn generate_square_beta_values(n_x: usize, n_y: usize) -> impl Iterator<Item = [DartIdType; 3]> {
101    // this loop hierarchy yields the value in correct order
102    // left to right first, then bottom to top
103    (0..n_y).flat_map(move |iy| {
104        (0..n_x).flat_map(move |ix| {
105                let d1 = (1 + 4 * ix + n_x * 4 * iy) as DartIdType;
106                let (d2, d3, d4) = (d1 + 1, d1 + 2, d1 + 3);
107                // beta images of [d1, d2, d3, d4]
108                [
109                    [ d4, d2, if iy == 0     { 0 } else { d3 - 4 * n_x as DartIdType } ],
110                    [ d1, d3, if ix == n_x-1 { 0 } else { d2 + 6                     } ],
111                    [ d2, d4, if iy == n_y-1 { 0 } else { d1 + 4 * n_x as DartIdType } ],
112                    [ d3, d1, if ix == 0     { 0 } else { d4 - 6                     } ],
113                ]
114                .into_iter()
115            })
116        })
117}

235/// Internal grid-building routine
236#[allow(clippy::cast_possible_truncation, clippy::too_many_lines)]
237pub(crate) fn build_3d_grid<T: CoordsFloat>(
238    builder: CMapBuilder<3>,
239    origin: Vertex3<T>,
240    n_cells_per_axis: [usize; 3],
241    lengths: [T; 3],
242) -> CMap3<T> {
243    let [n_square_x, n_square_y, n_square_z] = n_cells_per_axis;
244    let n_darts = 24 * n_square_x * n_square_y * n_square_z;
245
246    let map: CMap3<T> = CMapBuilder::<3>::from_n_darts_and_attributes(n_darts, builder)
247        .build()
248        .unwrap();
249
250    // init beta functions
251    (1..=n_darts as DartIdType)
252        .zip(generate_hex_beta_values(n_cells_per_axis))
253        .par_bridge()
254        .for_each(|(dart, images)| {
255            map.set_betas(dart, images);
256        });
257
258    // place vertices
259    (1..=n_darts as DartIdType)
260        .into_par_iter()
261        .filter(|d| *d as VertexIdType == map.vertex_id(*d))
262        .for_each(|d| {
263            let v = origin + generate_hex_offset(d, n_cells_per_axis, lengths);
264            map.force_write_vertex(d as VertexIdType, v);
265        });
266
267    // check the number of built volumes
268    // this is set as debug only because the operation cost scales with map size
269    // this can quickly overshadow the exectime of all previous code
270    debug_assert_eq!(
271        map.iter_volumes().count(),
272        n_square_x * n_square_y * n_square_z
273    );
274
275    map
276}

295#[allow(clippy::cast_possible_truncation,clippy::inline_always)]
296#[rustfmt::skip]
297#[inline(always)]
298fn generate_hex_beta_values(
299    [n_x, n_y, n_z]: [usize; 3],
300) -> impl Iterator<Item = [DartIdType; 4]> {
301    // this loop hierarchy yields the value in correct order
302    // left to right first, then bottom to top
303    (0..n_z).flat_map(move |iz| {
304        (0..n_y).flat_map(move |iy| {
305            (0..n_x).flat_map(move |ix| {
306                let d1 = (1 + 24 * ix + n_x * 24 * iy + n_x * n_y * 24 * iz) as DartIdType;
307                let (    d2 , d3 , d4 , d5 , d6 , d7 , d8 ,
308                    d9 , d10, d11, d12, d13, d14, d15, d16,
309                    d17, d18, d19, d20, d21, d22, d23, d24,
310                ) = (        d1 + 1 , d1 + 2 , d1 + 3 , d1 + 4 , d1 + 5 , d1 + 6 , d1 + 7 ,
311                    d1 + 8 , d1 + 9 , d1 + 10, d1 + 11, d1 + 12, d1 + 13, d1 + 14, d1 + 15,
312                    d1 + 16, d1 + 17, d1 + 18, d1 + 19, d1 + 20, d1 + 21, d1 + 22, d1 + 23,
313                );
314                let noffset_x = 24;
315                let noffset_y = noffset_x * n_x as DartIdType;
316                let noffset_z = noffset_y * n_y  as DartIdType;
317
318                // beta images of the cube (tm)
319                [
320                    // down (1, y-)
321                    [d4 , d2 , d5 , if iy == 0       { 0 } else { d21 - noffset_y }],
322                    [d1 , d3 , d9 , if iy == 0       { 0 } else { d24 - noffset_y }],
323                    [d2 , d4 , d13, if iy == 0       { 0 } else { d23 - noffset_y }],
324                    [d3 , d1 , d17, if iy == 0       { 0 } else { d22 - noffset_y }],
325                    // side (5 , z-)
326                    [d8 , d6 , d1 , if iz == 0       { 0 } else { d13 - noffset_z }],
327                    [d5 , d7 , d20, if iz == 0       { 0 } else { d16 - noffset_z }],
328                    [d6 , d8 , d21, if iz == 0       { 0 } else { d15 - noffset_z }],
329                    [d7 , d5 , d10, if iz == 0       { 0 } else { d14 - noffset_z }],
330                    // side (9 , x+)
331                    [d12, d10, d2 , if ix == n_x - 1 { 0 } else { d17 + noffset_x }],
332                    [d9 , d11, d8 , if ix == n_x - 1 { 0 } else { d20 + noffset_x }],
333                    [d10, d12, d24, if ix == n_x - 1 { 0 } else { d19 + noffset_x }],
334                    [d11, d9 , d14, if ix == n_x - 1 { 0 } else { d18 + noffset_x }],
335                    // side (13, z+)
336                    [d16, d14, d3 , if iz == n_z - 1 { 0 } else { d5  + noffset_z }],
337                    [d13, d15, d12, if iz == n_z - 1 { 0 } else { d8  + noffset_z }],
338                    [d14, d16, d23, if iz == n_z - 1 { 0 } else { d7  + noffset_z }],
339                    [d15, d13, d18, if iz == n_z - 1 { 0 } else { d6  + noffset_z }],
340                    // side (17, x-)
341                    [d20, d18, d4 , if ix == 0       { 0 } else { d9  - noffset_x }],
342                    [d17, d19, d16, if ix == 0       { 0 } else { d12 - noffset_x }],
343                    [d18, d20, d22, if ix == 0       { 0 } else { d11 - noffset_x }],
344                    [d19, d17, d6 , if ix == 0       { 0 } else { d10 - noffset_x }],
345                    // up   (21, y+)
346                    [d24, d22, d7 , if iy == n_y - 1 { 0 } else { d1  + noffset_y }],
347                    [d21, d23, d19, if iy == n_y - 1 { 0 } else { d4  + noffset_y }],
348                    [d22, d24, d15, if iy == n_y - 1 { 0 } else { d3  + noffset_y }],
349                    [d23, d21, d11, if iy == n_y - 1 { 0 } else { d2  + noffset_y }],
350                ]
351                .into_iter()
352            })
353        })
354    })
355}

357#[allow(
358    clippy::cast_possible_truncation,
359    clippy::inline_always,
360    clippy::match_same_arms,
361    clippy::too_many_lines,
362    clippy::many_single_char_names
363)]
364#[inline(always)]
365fn generate_hex_offset<T: CoordsFloat>(
366    dart: DartIdType,
367    [n_x, n_y, _]: [usize; 3],
368    [lx, ly, lz]: [T; 3],
369) -> Vector3<T> {
370    // d = p + 24*x + 24*NX*y + 24*NX*NY*z
371    let d = dart as usize;
372    let dm = d % 24;
373    let dmm = d % (24 * n_x);
374    let dmmm = d % (24 * n_x * n_y);
375    let p = dm;
376    let x = (dmm - dm) / 24;
377    let y = (dmmm - dmm) / (24 * n_x);
378    let z = (d - dmmm) / (24 * n_x * n_y);
379    match p {
380        // d1 to d24
381        // y- face
382        1 | 6 | 17 => Vector3(
383            T::from(x).unwrap() * lx,
384            T::from(y).unwrap() * ly,
385            T::from(z).unwrap() * lz,
386        ),
387        2 | 5 | 10 => Vector3(
388            T::from(x + 1).unwrap() * lx,
389            T::from(y).unwrap() * ly,
390            T::from(z).unwrap() * lz,
391        ),
392        3 | 9 | 14 => Vector3(
393            T::from(x + 1).unwrap() * lx,
394            T::from(y).unwrap() * ly,
395            T::from(z + 1).unwrap() * lz,
396        ),
397        4 | 13 | 18 => Vector3(
398            T::from(x).unwrap() * lx,
399            T::from(y).unwrap() * ly,
400            T::from(z + 1).unwrap() * lz,
401        ),
402        7 | 20 | 22 => Vector3(
403            T::from(x).unwrap() * lx,
404            T::from(y + 1).unwrap() * ly,
405            T::from(z).unwrap() * lz,
406        ),
407        8 | 11 | 21 => Vector3(
408            T::from(x + 1).unwrap() * lx,
409            T::from(y + 1).unwrap() * ly,
410            T::from(z).unwrap() * lz,
411        ),
412        12 | 15 | 0 => Vector3(
413            T::from(x + 1).unwrap() * lx,
414            T::from(y + 1).unwrap() * ly,
415            T::from(z + 1).unwrap() * lz,
416        ),
417        16 | 19 | 23 => Vector3(
418            T::from(x).unwrap() * lx,
419            T::from(y + 1).unwrap() * ly,
420            T::from(z + 1).unwrap() * lz,
421        ),
422        _ => unreachable!(),
423    }
424}

650fn generate_tet_offset<T: CoordsFloat>(
651    dart: DartIdType,
652    [n_x, n_y, _]: [usize; 3],
653    [lx, ly, lz]: [T; 3],
654) -> Vector3<T> {
655    // d = p + 24*x + 24*NX*y + 24*NX*NY*z
656    let d = dart as usize;
657    let dm = d % 60;
658    let dmm = d % (60 * n_x);
659    let dmmm = d % (60 * n_x * n_y);
660    let p = dm;
661    let x = (dmm - dm) / 60;
662    let y = (dmmm - dmm) / (60 * n_x);
663    let z = (d - dmmm) / (60 * n_x * n_y);
664    let mirror = ((x + y + z) & 1) == 0; // is even
665    if mirror {
666        match p {
667            1 | 5 | 10 => Vector3(
668                T::from(x).unwrap() * lx,
669                T::from(y).unwrap() * ly,
