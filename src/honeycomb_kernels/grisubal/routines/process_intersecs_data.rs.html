17pub(crate) fn group_intersections_per_edge<T: CoordsFloat>(
18    cmap: &mut CMap2<T>,
19    intersection_metadata: Vec<(DartIdType, T)>,
20) -> (IntersectionsPerEdge<T>, DartSlices) {
21    // group intersection data per edge, and associate an ID to each
22    let mut edge_intersec: HashMap<EdgeIdType, Vec<(usize, T, DartIdType)>> = HashMap::default();
23    intersection_metadata
24        .into_iter()
25        .filter(|(_, t)| !t.is_nan())
26        .enumerate()
27        .for_each(|(idx, (dart_id, mut t))| {
28            // classify intersections per edge_id & adjust t if  needed
29            let edge_id = cmap.edge_id(dart_id);
30            // condition works in 2D because edges are 2 darts at most
31            if edge_id != dart_id {
32                t = T::one() - t;
33            }
34            if let Some(storage) = edge_intersec.get_mut(&edge_id) {
35                // not the first intersection with this given edge
36                storage.push((idx, t, dart_id));
37            } else {
38                // first intersection with this given edge
39                edge_intersec.insert(edge_id, vec![(idx, t, dart_id)]);
40            }
41        });
42
43    // sort per t for later
44    for vs in edge_intersec.values_mut() {
45        // panic unreachable because t s.t. t == NaN have been filtered previously
46        vs.sort_by(|(_, t1, _), (_, t2, _)| t1.partial_cmp(t2).expect("E: unreachable"));
47    }
48
49    // prealloc darts that will be used for vertex insertion
50    let n_darts_per_seg: Vec<_> = edge_intersec.values().map(|vs| 2 * vs.len()).collect();
51    let n_tot: usize = n_darts_per_seg.iter().sum();
52    let tmp = cmap.allocate_used_darts(n_tot) as usize;
53    // the prefix sum gives an offset that corresponds to the starting index of each slice, minus
54    // the location of the allocated dart block (given by `tmp`)
55    // end of the slice is deduced using these values and the number of darts the current seg needs
56    let prefix_sum: Vec<usize> = n_darts_per_seg
57        .iter()
58        .scan(0, |state, &n_d| {
59            *state += n_d;
60            Some(*state - n_d) // we want an offset, not the actual sum
61        })
62        .collect();
63
64    #[allow(clippy::cast_possible_truncation)]
65    let dart_slices: Vec<Vec<DartIdType>> = n_darts_per_seg
66        .iter()
67        .zip(prefix_sum.iter())
68        .map(|(n_d, start)| {
69            ((tmp + start) as DartIdType..(tmp + start + n_d) as DartIdType).collect::<Vec<_>>()
70        })
71        .collect();
72
73    (edge_intersec, dart_slices)
74}

76pub(crate) fn compute_intersection_ids<T: CoordsFloat>(
77    n_intersec: usize,
78    edge_intersec: &IntersectionsPerEdge<T>,
79    dart_slices: &DartSlices,
80) -> Vec<DartIdType> {
81    let mut res = vec![NULL_DART_ID; n_intersec];
82    for ((edge_id, vs), new_darts) in edge_intersec.iter().zip(dart_slices.iter()) {
83        // order should be consistent between collection because of the sort_by call
84        let hl = new_darts.len() / 2; // half-length; also equal to n_intermediate
85        let fh = &new_darts[..hl]; // first half;  used for the side of edge id
86        let sh = &new_darts[hl..]; // second half; used for the opposite side
87        for (i, (id, _, old_dart_id)) in vs.iter().enumerate() {
88            // readjust according to intersection side
89            res[*id] = if *old_dart_id == *edge_id {
90                fh[i]
91            } else {
92                sh[hl - 1 - i]
93            };
94        }
95    }
96    res
