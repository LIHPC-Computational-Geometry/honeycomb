13#[allow(clippy::cast_possible_truncation)]
14pub(crate) fn insert_edges_in_map<T: CoordsFloat>(cmap: &mut CMap2<T>, edges: &[MapEdge<T>]) {
15    let dart_slices = build_workload(cmap, edges);
16
17    for (
18        i,
19        (
20            MapEdge {
21                start,
22                intermediates,
23                end,
24            },
25            dslice,
26        ),
27    ) in edges.iter().zip(dart_slices.iter()).enumerate()
28    {
29        let &[d_new, b2_d_new] = &dslice[0..2] else {
30            unreachable!()
31        };
32        build_base_edge(cmap, *start, *end, [d_new, b2_d_new]);
33
34        if !intermediates.is_empty() {
35            // create the topology components
36            let edge_id = cmap.edge_id(d_new);
37            let new_darts = &dslice[2..];
38            atomically_with_err(|t| {
39                insert_vertices_on_edge(
40                    cmap,
41                    t,
42                    edge_id,
43                    new_darts,
44                    &vec![T::from(0.5).unwrap(); intermediates.len()],
45                )
46            })
47            .unwrap();
48            // replace placeholder vertices
49            let mut dart_id = cmap.beta::<1>(edge_id as DartIdType);
50            for v in intermediates {
51                let vid = cmap.vertex_id(dart_id);
52                let _ = cmap.force_write_vertex(vid, *v);
53                if cmap.contains_attribute::<VertexAnchor>() {
54                    let _ = cmap
55                        .force_write_attribute::<VertexAnchor>(vid, VertexAnchor::Node(i as u32));
56                }
57                dart_id = cmap.beta::<1>(dart_id);
58            }
59        }
60
61        mark_boundary(cmap, *start, *end);
62    }
63}

65#[allow(clippy::cast_possible_truncation)]
66fn build_workload<T: CoordsFloat>(
67    cmap: &mut CMap2<T>,
68    edges: &[MapEdge<T>],
69) -> Vec<Vec<DartIdType>> {
70    // allocate all darts needed
71    let n_tot: usize = edges.iter().map(|e| 2 + 2 * e.intermediates.len()).sum();
72    let tmp = cmap.allocate_used_darts(n_tot) as usize;
73
74    // the prefix sum gives an offset that corresponds to the starting index of each slice, minus
75    // the location of the allocated dart block (given by `tmp`)
76    // we can deduce theend of the slice from start index (prefix sum) + number of darts
77    edges
78        .iter()
79        .map(|e| 2 + 2 * e.intermediates.len())
80        .scan(0, |state, n_d| {
81            *state += n_d;
82            Some((n_d, *state - n_d)) // we want an offset, not the actual sum
83        })
84        .map(|(n_d, start)| {
85            ((tmp + start) as DartIdType..(tmp + start + n_d) as DartIdType).collect::<Vec<_>>()
86        })
87        .collect()
88}

90fn build_base_edge<T: CoordsFloat>(
91    cmap: &CMap2<T>,
92    start: DartIdType,
93    end: DartIdType,
94    [d_new, b2_d_new]: [DartIdType; 2],
95) {
96    // remove deprecated connectivities & save what data is necessary
97    let b1_start_old = cmap.beta::<1>(start);
98    let b0_end_old = cmap.beta::<0>(end);
99    cmap.force_unlink::<1>(start).unwrap();
100    cmap.force_unlink::<1>(b0_end_old).unwrap();
101
102    cmap.force_link::<2>(d_new, b2_d_new).unwrap();
103
104    // rebuild - this is the final construct if there are no intermediates
105    cmap.force_link::<1>(start, d_new).unwrap();
106    cmap.force_link::<1>(b2_d_new, b1_start_old).unwrap();
107    cmap.force_link::<1>(d_new, end).unwrap();
108    cmap.force_link::<1>(b0_end_old, b2_d_new).unwrap();
109}

111fn mark_boundary<T: CoordsFloat>(cmap: &CMap2<T>, start: DartIdType, end: DartIdType) {
112    let mut d_boundary = cmap.beta::<1>(start);
113    while d_boundary != end {
114        cmap.force_write_attribute::<Boundary>(d_boundary, Boundary::Left);
115        cmap.force_write_attribute::<Boundary>(cmap.beta::<2>(d_boundary), Boundary::Right);
116        d_boundary = cmap.beta::<1>(d_boundary);
117    }
118}
