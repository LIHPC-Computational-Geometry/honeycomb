12/// Check for orientation issue **per boundary**.
13///
14/// This function check for the most obvious orientation issue; given a boundary, are all segments making it up
15/// oriented consistently. If it is not the case, then there is at least one of:
16///
17/// - a vertex being the origin of two segment
18/// - a vertex being the end-point of two segment
19///
20/// This does not cover consistent orientation across distinct boundaries (e.g. a geometry with a hole in it).
21pub fn detect_orientation_issue<T: CoordsFloat>(
22    geometry: &Geometry2<T>,
23) -> Result<(), GrisubalError> {
24    let mut origins = HashSet::new();
25    let mut endpoints = HashSet::new();
26
27    for (orig, endp) in &geometry.segments {
28        if !origins.insert(orig) || !endpoints.insert(endp) {
29            return Err(GrisubalError::InconsistentOrientation(
30                "in-boundary inconsistency",
31            ));
32        }
33    }
34
35    Ok(())
36}

38#[allow(clippy::cast_precision_loss)]
39pub fn compute_overlapping_grid<T: CoordsFloat>(
40    geometry: &Geometry2<T>,
41    [len_cell_x, len_cell_y]: [T; 2],
42    keep_all_poi: bool,
43) -> Result<([usize; 2], Vertex2<T>), GrisubalError> {
44    // compute the minimum bounding box
45    let (mut min_x, mut max_x, mut min_y, mut max_y): (T, T, T, T) = {
46        let Some(tmp) = geometry.vertices.first() else {
47            return Err(GrisubalError::InvalidShape("no vertex in shape"));
48        };
49
50        (tmp.x(), tmp.x(), tmp.y(), tmp.y())
51    };
52
53    geometry.vertices.iter().for_each(|v| {
54        min_x = min_x.min(v.x());
55        max_x = max_x.max(v.x()); // may not be optimal
56        min_y = min_y.min(v.y()); // don't care
57        max_y = max_y.max(v.y());
58    });
59
60    if max_x <= min_x {
61        return Err(GrisubalError::InvalidShape(
62            "bounding values along X axis are equal",
63        ));
64    }
65    if max_y <= min_y {
66        return Err(GrisubalError::InvalidShape(
67            "bounding values along Y axis are equal",
68        ));
69    }
70
71    // compute characteristics of the overlapping Cartesian grid
72
73    // create a ~one-and-a-half cell buffer to contain the geometry
74    // this, along with the `+1` below, guarantees that
75    // dart at the boundary of the grid are not intersected by the geometry
76    let mut og_x = min_x - len_cell_x * T::from(1.5).unwrap();
77    let mut og_y = min_y - len_cell_y * T::from(1.5).unwrap();
78    // we check for some extremely annoying cases here
79    // if some are detected, the origin is incrementally shifted
80    let (mut on_corner, mut reflect) = detect_overlaps(
81        geometry,
82        [len_cell_x, len_cell_y],
83        Vertex2(og_x, og_y),
84        !keep_all_poi,
85    );
86    let mut i = 1;
87
88    while on_corner | reflect {
89        eprintln!(
90            "W: land on corner: {on_corner} - reflect on an axis: {reflect}, shifting origin"
91        );
92        og_x += len_cell_x * T::from(1. / (2_i32.pow(i + 1) as f32)).unwrap();
93        og_y += len_cell_y * T::from(1. / (2_i32.pow(i + 1) as f32)).unwrap();
94        (on_corner, reflect) = detect_overlaps(
95            geometry,
96            [len_cell_x, len_cell_y],
97            Vertex2(og_x, og_y),
98            !keep_all_poi,
99        );
100        i += 1;
101    }
102
103    let n_cells_x = ((max_x - og_x) / len_cell_x).ceil().to_usize().unwrap() + 1;
104    let n_cells_y = ((max_y - og_y) / len_cell_y).ceil().to_usize().unwrap() + 1;
105
106    Ok(([n_cells_x, n_cells_y], Vertex2(og_x, og_y)))
107}

109/// Remove from their geometry points of interest that intersect with a grid of specified dimension.
110///
111/// This function works under the assumption that the grid is Cartesian & has its origin on `(0.0, 0.0)`.
112pub fn remove_redundant_poi<T: CoordsFloat>(
113    geometry: &mut Geometry2<T>,
114    [cx, cy]: [T; 2],
115    origin: Vertex2<T>,
116) {
117    // PoI that land on the grid create a number of issues; removing them is ok since we're intersecting the grid
118    // at their coordinates, so the shape will be captured via intersection anyway
119    geometry.poi.retain(|idx| {
120        let v = geometry.vertices[*idx];
121        // origin is assumed to be (0.0, 0.0)
122        let on_x_axis = ((v.x() - origin.x()) % cx).is_zero();
123        let on_y_axis = ((v.y() - origin.y()) % cy).is_zero();
124        !(on_x_axis | on_y_axis)
125    });
126}

128#[allow(clippy::map_all_any_identity)]
129pub fn detect_overlaps<T: CoordsFloat>(
130    geometry: &Geometry2<T>,
131    [cx, cy]: [T; 2],
132    origin: Vertex2<T>,
133    overlap_only_corners: bool,
134) -> (bool, bool) {
135    let on_grid = geometry
136        .vertices
137        .iter()
138        .map(|v| {
139            let on_x_axis = ((v.x() - origin.x()) % cx).is_zero();
140            let on_y_axis = ((v.y() - origin.y()) % cy).is_zero();
141            if overlap_only_corners {
142                on_x_axis && on_y_axis
143            } else {
144                on_x_axis || on_y_axis
145            }
146        })
147        .any(|a| a);
148
149    let bad_reflection = geometry
150        .vertices
151        .iter()
152        .enumerate()
153        .filter_map(|(id, v)| {
154            let on_x_axis = ((v.x() - origin.x()) % cx).is_zero();
155            let on_y_axis = ((v.y() - origin.y()) % cy).is_zero();
156            if on_x_axis | on_y_axis {
157                return Some(id);
158            }
159            None
160        })
161        // skip vertices that do not belong to the boundary
162        .filter(|id| {
163            geometry
164                .segments
165                .iter()
166                .any(|(v1, v2)| (id == v1) || (id == v2))
167        })
168        .map(|id| {
169            // if a vertex appear in the boundary, there should be both a segment landing and a
170            // segment starting on the vertex; hence `.expect()`
171            let vid_in = geometry
172                .segments
173                .iter()
174                .find_map(|(vin, ref_id)| {
175                    if id == *ref_id {
176                        return Some(*vin);
177                    }
178                    None
179                })
180                .expect("E: found a vertex with no incident segment - is the geometry open?");
181            // same
182            let vid_out = geometry
183                .segments
184                .iter()
185                .find_map(|(ref_id, vout)| {
186                    if id == *ref_id {
187                        return Some(*vout);
188                    }
189                    None
190                })
191                .expect("E: found a vertex with no incident segment - is the geometry open?");
192            let v_in = geometry.vertices[vid_in];
193            let v_out = geometry.vertices[vid_out];
194            let Vertex2(ox, oy) = origin;
195            let (c_in, c_out) = (
196                GridCellId(
197                    ((v_in.x() - ox) / cx).floor().to_usize().unwrap(),
198                    ((v_in.y() - oy) / cy).floor().to_usize().unwrap(),
199                ),
200                GridCellId(
201                    ((v_out.x() - ox) / cx).floor().to_usize().unwrap(),
202                    ((v_out.y() - oy) / cy).floor().to_usize().unwrap(),
203                ),
204            );
205            // if v_in and v_out belong to the same grid cell, there was a "reflection" on one
206            // of the grid's axis
207            c_in == c_out
208        })
209        .any(|a| a);
210
211    (on_grid, bad_reflection)
212}
