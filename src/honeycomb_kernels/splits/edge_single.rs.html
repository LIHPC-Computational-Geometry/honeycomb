12/// Split an edge into two segments.
13///
14/// <div class="warning">
15/// This implementation is 2D specific.
16/// </div>
17///
18/// This method takes all darts of an edge and rebuild connections in order to create a new
19/// point on this edge. The position of the point defaults to the midway point, but it can
20/// optionally be specified.
21///
22/// In order to minimize editing of embedded data, the original darts are kept to their
23/// original vertices while the new darts are used to model the new point.
24///
25/// For an illustration of both principles, refer to the example.
26///
27/// # Arguments
28///
29/// - `cmap: &mut CMap2<T>` -- Reference to the modified map.
30/// - `edge_id: EdgeIdentifier` -- Edge to split in two.
31/// - `midpoint_vertex: Option<T>` -- Relative position of the new vertex, starting from the
32///   vertex of the dart sharing `edge_id` as its identifier.
33///
34/// # Return / Errors
35///
36/// This method will return:
37/// - `Ok(())` if the operation is successful & the edge was split
38/// - `Err(SplitEdgeError)` if the operation fails & the edge is left unchanged. Causes of failure
39///   are described in [`SplitEdgeError`]'s documentation.
40///
41/// # Example
42///
43/// Given an edge made of darts `1` and `2`, these darts respectively encoding vertices
44/// `(0.0, 0.0)` and `(2.0, 0.0)`, calling `map.split_edge(1, Some(0.2))` would result in the
45/// creation of two new darts, a new vertex (ID `3`) at position `(0.4, 0.0)` and the following
46/// topology:
47///
48/// ```text
49///    +----1---->              +-1-> +-3->     |
50///  1             2    =>    1      3      2   | + denote darts that encode vertex IDs
51///    <----2----+              <-4-- <-2-+     |
52/// ```
53pub fn split_edge<T: CoordsFloat>(
54    cmap: &mut CMap2<T>,
55    edge_id: EdgeIdType,
56    midpoint_vertex: Option<T>,
57) -> Result<(), SplitEdgeError> {
58    // midpoint check
59    if midpoint_vertex.is_some_and(|t| (t >= T::one()) | (t <= T::zero())) {
60        return Err(SplitEdgeError::VertexBound);
61    }
62
63    // base darts making up the edge
64    let base_dart1 = edge_id as DartIdType;
65    let base_dart2 = cmap.beta::<2>(base_dart1);
66
67    let new_darts = if base_dart2 == NULL_DART_ID {
68        (cmap.add_free_dart(), NULL_DART_ID)
69    } else {
70        let tmp = cmap.add_free_darts(2);
71        (tmp, tmp + 1)
72    };
73
74    atomically_with_err(|trans| inner_split(cmap, trans, base_dart1, new_darts, midpoint_vertex))
75}

78/// Split an edge into two segments.
79///
80/// <div class="warning">
81/// This implementation is 2D specific.
82/// </div>
83///
84/// This method is a variant of [`split_edge`] where inline dart allocations are removed. The
85/// aim of this variant is to enhance performance by enabling the user to pre-allocate a number
86/// of darts.
87///
88/// The method follows the same logic as the regular [`split_edge`], the only difference being
89/// that the new darts won't be added to the map on the fly. Instead, the method uses the two
90/// darts passed as argument (`new_darts`) to build the new segments. Consequently, there is no
91/// guarantee that IDs will be consistent between this and the regular method.
92///
93/// # Arguments
94///
95/// - `cmap: &mut CMap2<T>` -- Reference to the modified map.
96/// - `edge_id: EdgeIdentifier` -- Edge to split in two.
97/// - `new_darts: (DartIdentifier, DartIdentifier)` -- Dart IDs used to build the new segments.
98/// - `midpoint_vertex: Option<T>` -- Relative position of the new vertex, starting from the
99///   vertex of the dart sharing `edge_id` as its identifier.
100///
101/// ## Dart IDs Requirements & Usage
102///
103/// Because of the dimension, the number of dart needed to perform this operation is at most
104/// two. These are the requirements for these two darts:
105/// - identifiers are passed as a tuple.
106/// - the first dart of the tuple will always be used if the operation is successful.
107/// - the second dart of the tuple will only be used if the original edge is made of two darts;
108///   if that is not the case, the second dart ID can be `NULL_DART_ID`.
109/// - both of these darts should be free
110///
111/// # Return / Errors
112///
113/// This method will return:
114/// - `Ok(())` if the operation is successful & the edge was split
115/// - `Err(SplitEdgeError)` if the operation fails & the edge is left unchanged. Causes of failure
116///   are described in [`SplitEdgeError`]'s documentation and in requirements mentionned above.
117pub fn split_edge_transac<T: CoordsFloat>(
118    cmap: &CMap2<T>,
119    trans: &mut Transaction,
120    edge_id: EdgeIdType,
121    new_darts: (DartIdType, DartIdType), // 2D => statically known number of darts
122    midpoint_vertex: Option<T>,
123) -> TransactionClosureResult<(), SplitEdgeError> {
124    // midpoint check
125    if midpoint_vertex.is_some_and(|t| (t >= T::one()) | (t <= T::zero())) {
126        abort(SplitEdgeError::VertexBound)?;
127    }
128
129    // base darts making up the edge
130    let base_dart1 = edge_id as DartIdType;
131    let base_dart2 = cmap.beta_transac::<2>(trans, base_dart1)?;
132
133    // FIXME: is_free should be transactional
134    if new_darts.0 == NULL_DART_ID || !cmap.is_free(new_darts.0) {
135        abort(SplitEdgeError::InvalidDarts(
136            "first dart is null or not free",
137        ))?;
138    }
139    // FIXME: is_free should be transactional
140    if base_dart2 != NULL_DART_ID && (new_darts.1 == NULL_DART_ID || !cmap.is_free(new_darts.1)) {
141        abort(SplitEdgeError::InvalidDarts(
142            "second dart is null or not free",
143        ))?;
144    }
145
146    inner_split(cmap, trans, base_dart1, new_darts, midpoint_vertex)
147}

149// --- common inner routine
150
151fn inner_split<T: CoordsFloat>(
152    cmap: &CMap2<T>,
153    trans: &mut Transaction,
154    base_dart1: DartIdType,
155    new_darts: (DartIdType, DartIdType), // 2D => statically known number of darts
156    midpoint_vertex: Option<T>,
157) -> TransactionClosureResult<(), SplitEdgeError> {
158    // base darts making up the edge
159    let base_dart2 = cmap.beta_transac::<2>(trans, base_dart1)?;
160    if base_dart2 == NULL_DART_ID {
161        let b1d1_old = cmap.beta_transac::<1>(trans, base_dart1)?;
162        let b1d1_new = new_darts.0;
163        let (vid1, vid2) = (
164            cmap.vertex_id_transac(trans, base_dart1)?,
165            cmap.vertex_id_transac(trans, b1d1_old)?,
166        );
167        let (Some(v1), Some(v2)) = (
168            cmap.read_vertex(trans, vid1)?,
169            cmap.read_vertex(trans, vid2)?,
170        ) else {
171            abort(SplitEdgeError::UndefinedEdge)?
172        };
173        // unsew current dart
174        if b1d1_old != NULL_DART_ID {
175            try_or_coerce!(cmap.unlink::<1>(trans, base_dart1), SplitEdgeError);
176        }
177        // cmap.set_beta::<1>(base_dart1, 0);
178        // cmap.set_beta::<0>(b1d1_old, 0);
179        // rebuild the edge
180        try_or_coerce!(cmap.link::<1>(trans, base_dart1, b1d1_new), SplitEdgeError);
181        try_or_coerce!(cmap.link::<1>(trans, b1d1_new, b1d1_old), SplitEdgeError);
182        // insert the new vertex
183        let seg = v2 - v1;
184        let vnew = cmap.vertex_id_transac(trans, b1d1_new)?;
185        cmap.write_vertex(
186            trans,
187            vnew,
188            midpoint_vertex.map_or(Vertex2::average(&v1, &v2), |t| v1 + seg * t),
189        )?;
190        Ok(())
191    } else {
192        let b1d1_old = cmap.beta_transac::<1>(trans, base_dart1)?;
193        let b1d2_old = cmap.beta_transac::<1>(trans, base_dart2)?;
194        let (b1d1_new, b1d2_new) = new_darts;
195        let (vid1, vid2) = (
196            cmap.vertex_id_transac(trans, base_dart1)?,
197            cmap.vertex_id_transac(trans, base_dart2)?,
198        );
199        let (Some(v1), Some(v2)) = (
200            cmap.read_vertex(trans, vid1)?,
201            cmap.read_vertex(trans, vid2)?,
202        ) else {
203            abort(SplitEdgeError::UndefinedEdge)?
204        };
205        // unsew current darts
206        if b1d1_old != NULL_DART_ID {
207            try_or_coerce!(cmap.unlink::<1>(trans, base_dart1), SplitEdgeError);
208        }
209        if b1d2_old != NULL_DART_ID {
210            try_or_coerce!(cmap.unlink::<1>(trans, base_dart2), SplitEdgeError);
211        }
212        // cmap.set_beta::<1>(base_dart1, 0);
213        // cmap.set_beta::<0>(b1d1_old, 0);
214        // cmap.set_beta::<1>(base_dart2, 0);
215        // cmap.set_beta::<0>(b1d2_old, 0);
216        try_or_coerce!(cmap.unlink::<2>(trans, base_dart1), SplitEdgeError);
217        // rebuild the edge
218        try_or_coerce!(cmap.link::<1>(trans, base_dart1, b1d1_new), SplitEdgeError);
219        if b1d1_old != NULL_DART_ID {
220            try_or_coerce!(cmap.link::<1>(trans, b1d1_new, b1d1_old), SplitEdgeError);
221        }
222        try_or_coerce!(cmap.link::<1>(trans, base_dart2, b1d2_new), SplitEdgeError);
223        if b1d2_old != NULL_DART_ID {
224            try_or_coerce!(cmap.link::<1>(trans, b1d2_new, b1d2_old), SplitEdgeError);
225        }
226        try_or_coerce!(cmap.link::<2>(trans, base_dart1, b1d2_new), SplitEdgeError);
227        try_or_coerce!(cmap.link::<2>(trans, base_dart2, b1d1_new), SplitEdgeError);
228        // insert the new vertex
229        let seg = v2 - v1;
230        let vnew = cmap.vertex_id_transac(trans, b1d1_new)?;
231        cmap.write_vertex(
232            trans,
233            vnew,
234            midpoint_vertex.map_or(Vertex2::average(&v1, &v2), |t| v1 + seg * t),
235        )?;
236        Ok(())
237    }
