23pub(crate) fn parse_meta(meta_line: &str) -> Result<(String, usize, usize), BuilderError> {
24    let parts: Vec<&str> = meta_line.split_whitespace().collect();
25    if parts.len() != 3 {
26        return Err(BuilderError::BadMetaData("incorrect format"));
27    }
28
29    Ok((
30        parts[0].to_string(),
31        parts[1]
32            .parse()
33            .map_err(|_| BuilderError::BadMetaData("could not parse dimension"))?,
34        parts[2]
35            .parse()
36            .map_err(|_| BuilderError::BadMetaData("could not parse dart number"))?,
37    ))
38}

40impl TryFrom<String> for CMapFile {
41    type Error = BuilderError;
42
43    fn try_from(value: String) -> Result<Self, Self::Error> {
44        let mut sections = HashMap::default();
45        let mut current_section = String::new();
46
47        for line in value.trim().lines() {
48            let trimmed = line.trim();
49            if trimmed.is_empty() || trimmed.starts_with('#') {
50                // ignore empty & comment lines
51                continue;
52            }
53            if trimmed.starts_with('[') && trimmed.contains(']') {
54                // process section header
55                let section_name = trimmed.trim_matches(['[', ']']).to_lowercase();
56
57                if section_name != "meta"
58                    && section_name != "betas"
59                    && section_name != "unused"
60                    && section_name != "vertices"
61                {
62                    return Err(BuilderError::UnknownHeader(section_name));
63                }
64
65                if sections
66                    .insert(section_name.clone(), String::new())
67                    .is_some()
68                {
69                    return Err(BuilderError::DuplicatedSection(section_name));
70                }
71                current_section = section_name;
72
73                continue;
74            }
75            if !current_section.is_empty() {
76                // regular line
77                let line_without_comment = trimmed.split('#').next().unwrap().trim();
78                if !line_without_comment.is_empty() {
79                    let current_content = sections.get_mut(&current_section).unwrap();
80                    if !current_content.is_empty() {
81                        current_content.push('\n');
82                    }
83                    current_content.push_str(line_without_comment);
84                }
85            }
86        }
87
88        if !sections.contains_key("meta") {
89            // missing required section
90            return Err(BuilderError::MissingSection("meta"));
91        }
92        if !sections.contains_key("betas") {
93            // missing required section
94            return Err(BuilderError::MissingSection("betas"));
95        }
96
97        Ok(Self {
98            meta: parse_meta(sections["meta"].as_str())?,
99            betas: sections["betas"].clone(),
100            unused: sections.get("unused").cloned(),
101            vertices: sections.get("vertices").cloned(),
102        })
103    }
104}

108pub fn build_2d_from_cmap_file<T: CoordsFloat>(
109    f: CMapFile,
110    manager: AttrStorageManager, // FIXME: find a cleaner solution to populate the manager
111) -> Result<CMap2<T>, BuilderError> {
112    if f.meta.1 != 2 {
113        // mismatched dim
114        return Err(BuilderError::BadMetaData(
115            "mismatch between requested dimension and header",
116        ));
117    }
118    let map = CMap2::new_with_undefined_attributes(f.meta.2, manager);
119
120    // putting it in a scope to drop the data
121    let betas = f.betas.lines().collect::<Vec<_>>();
122    if betas.len() != 3 {
123        // mismatched dim
124        return Err(BuilderError::InconsistentData(
125            "wrong number of beta functions",
126        ));
127    }
128    let b0 = betas[0]
129        .split_whitespace()
130        .map(str::parse)
131        .collect::<Vec<_>>();
132    let b1 = betas[1]
133        .split_whitespace()
134        .map(str::parse)
135        .collect::<Vec<_>>();
136    let b2 = betas[2]
137        .split_whitespace()
138        .map(str::parse)
139        .collect::<Vec<_>>();
140
141    // mismatched dart number
142    if b0.len() != f.meta.2 + 1 {
143        return Err(BuilderError::InconsistentData(
144            "wrong number of values for the beta 0 function",
145        ));
146    }
147    if b1.len() != f.meta.2 + 1 {
148        return Err(BuilderError::InconsistentData(
149            "wrong number of values for the beta 1 function",
150        ));
151    }
152    if b2.len() != f.meta.2 + 1 {
153        return Err(BuilderError::InconsistentData(
154            "wrong number of values for the beta 2 function",
155        ));
156    }
157
158    for (d, b0d, b1d, b2d) in multizip((
159        (1..=f.meta.2),
160        b0.into_iter().skip(1),
161        b1.into_iter().skip(1),
162        b2.into_iter().skip(1),
163    )) {
164        let b0d = b0d.map_err(|_| BuilderError::BadValue("could not parse a b0 value"))?;
165        let b1d = b1d.map_err(|_| BuilderError::BadValue("could not parse a b1 value"))?;
166        let b2d = b2d.map_err(|_| BuilderError::BadValue("could not parse a b2 value"))?;
167        map.set_betas(d as DartIdType, [b0d, b1d, b2d]);
168    }
169
170    if let Some(unused) = f.unused {
171        for u in unused.split_whitespace() {
172            let d = u
173                .parse()
174                .map_err(|_| BuilderError::BadValue("could not parse an unused ID"))?;
175            map.release_dart(d)
176                .expect("E: unused dart has non-null beta images");
177        }
178    }
179
180    if let Some(vertices) = f.vertices {
181        for l in vertices.trim().lines() {
182            let mut it = l.split_whitespace();
183            let id: VertexIdType = it
184                .next()
185                .ok_or(BuilderError::BadValue("incorrect vertex line format"))?
186                .parse()
187                .map_err(|_| BuilderError::BadValue("could not parse vertex ID"))?;
188            let x: f64 = it
189                .next()
190                .ok_or(BuilderError::BadValue("incorrect vertex line format"))?
191                .parse()
192                .map_err(|_| BuilderError::BadValue("could not parse vertex x coordinate"))?;
193            let y: f64 = it
194                .next()
195                .ok_or(BuilderError::BadValue("incorrect vertex line format"))?
196                .parse()
197                .map_err(|_| BuilderError::BadValue("could not parse vertex y coordinate"))?;
198            if it.next().is_some() {
199                return Err(BuilderError::BadValue("incorrect vertex line format"));
200            }
201            map.force_write_vertex(id, (T::from(x).unwrap(), T::from(y).unwrap()));
202        }
203    }
204
205    Ok(map)
206}

280    }
281
282    if let Some(unused) = f.unused {
283        for u in unused.split_whitespace() {
284            let d = u
285                .parse()
286                .map_err(|_| BuilderError::BadValue("could not parse an unused ID"))?;
287            map.release_dart(d)
288                .expect("E: unused dart has non-null beta images");
289        }
290    }
291
292    if let Some(vertices) = f.vertices {
293        for l in vertices.trim().lines() {
294            let mut it = l.split_whitespace();
295            let id: VertexIdType = it
296                .next()
297                .ok_or(BuilderError::BadValue("incorrect vertex line format"))?
298                .parse()
299                .map_err(|_| BuilderError::BadValue("could not parse vertex ID"))?;
300            let x: f64 = it
301                .next()
302                .ok_or(BuilderError::BadValue("incorrect vertex line format"))?
303                .parse()
304                .map_err(|_| BuilderError::BadValue("could not parse vertex x coordinate"))?;
305            let y: f64 = it
306                .next()
307                .ok_or(BuilderError::BadValue("incorrect vertex line format"))?
308                .parse()
309                .map_err(|_| BuilderError::BadValue("could not parse vertex y coordinate"))?;
310            let z: f64 = it
311                .next()
312                .ok_or(BuilderError::BadValue("incorrect vertex line format"))?
313                .parse()
314                .map_err(|_| BuilderError::BadValue("could not parse vertex z coordinate"))?;
315            if it.next().is_some() {
316                return Err(BuilderError::BadValue("incorrect vertex line format"));
317            }
318            map.force_write_vertex(
319                id,
320                (
321                    T::from(x).unwrap(),
322                    T::from(y).unwrap(),
323                    T::from(z).unwrap(),
324                ),
325            );
326        }
327    }
328
329    Ok(map)
330}

334macro_rules! if_predicate_return_err {
335    ($pr: expr, $er: expr) => {
336        if $pr {
337            return Err($er);
338        }
339    };
340}
341
342macro_rules! build_vertices {
343    ($v: ident) => {{
344        if_predicate_return_err!(
345            !($v.len() % 3).is_zero(),
346            BuilderError::BadVtkData("vertex list contains an incomplete tuple")
347        );
348        $v.chunks_exact(3)
349            .map(|slice| {
350                // WE IGNORE Z values
351                let &[x, y, _] = slice else { unreachable!() };
352                Vertex2(T::from(x).unwrap(), T::from(y).unwrap())
353            })
354            .collect()
355    }};
356}

360#[allow(clippy::too_many_lines)]
361/// Internal building routine for [`CMap2::from_vtk_file`].
362///
363/// # Result / Errors
364///
365/// This implementation support only a very specific subset of VTK files. This result in many
366/// possibilities for failure. This function may return:
367///
368/// - `Ok(CMap2)` -- The file was successfully parsed and its content made into a 2-map.
369/// - `Err(BuilderError)` -- The function failed for one of the following reasons (sorted
370///   by [`BuilderError`] variants):
371///     - `UnsupportedVtkData`: The file contains unsupported data, i.e.:
372///         - file format isn't Legacy,
373///         - data set is something other than `UnstructuredGrid`,
374///         - coordinate representation type isn't `float` or `double`
375///         - mesh contains unsupported cell types (`PolyVertex`, `PolyLine`, `TriangleStrip`,
376///           `Pixel` or anything 3D)
377///     - `InvalidVtkFile`: The file contains inconsistencies, i.e.:
378///         - the number of coordinates cannot be divided by `3`, meaning a tuple is incomplete
379///         - the number of `Cells` and `CellTypes` isn't equal
380///         - a given cell has an inconsistent number of vertices with its specified cell type
381pub fn build_2d_from_vtk<T: CoordsFloat>(
382    value: Vtk,
383    mut _manager: AttrStorageManager, // FIXME: find a cleaner solution to populate the manager
384) -> Result<CMap2<T>, BuilderError> {
385    let mut cmap: CMap2<T> = CMap2::new(0);
386    let mut sew_buffer: BTreeMap<(usize, usize), DartIdType> = BTreeMap::new();
387    match value.data {
388        DataSet::ImageData { .. }
389        | DataSet::StructuredGrid { .. }
390        | DataSet::RectilinearGrid { .. }
391        | DataSet::PolyData { .. }
392        | DataSet::Field { .. } => {
393            return Err(BuilderError::UnsupportedVtkData("dataset not supported"));
394        }
395        DataSet::UnstructuredGrid { pieces, .. } => {
396            let mut tmp = pieces.iter().map(|piece| {
397                // assume inline data
398                let Ok(tmp) = piece.load_piece_data(None) else {
399                    return Err(BuilderError::UnsupportedVtkData("not inlined data piece"));
400                };
401
402                // build vertex list
403                // since we're expecting coordinates, we'll assume floating type
404                // we're also converting directly to our vertex type since we're building a 2-map
405                let vertices: Vec<Vertex2<T>> = match tmp.points {
406                    IOBuffer::F64(v) => build_vertices!(v),
407                    IOBuffer::F32(v) => build_vertices!(v),
408                    _ => {
409                        return Err(BuilderError::UnsupportedVtkData(
410                            "unsupported coordinate type",
411                        ));
412                    }
413                };
414
415                let vtkio::model::Cells { cell_verts, types } = tmp.cells;
416                match cell_verts {
417                    VertexNumbers::Legacy {
418                        num_cells,
419                        vertices: verts,
420                    } => {
421                        // check basic stuff
422                        if_predicate_return_err!(
423                            num_cells as usize != types.len(),
424                            BuilderError::BadVtkData("different # of cell in CELLS and CELL_TYPES")
425                        );
426
427                        // build a collection of vertex lists corresponding of each cell
428                        let mut cell_components: Vec<Vec<usize>> = Vec::new();
429                        let mut take_next = 0;
430                        for vertex_id in &verts {
431                            if take_next.is_zero() {
432                                // making it usize since it's a counter
433                                take_next = *vertex_id as usize;
434                                cell_components.push(Vec::with_capacity(take_next));
435                            } else {
436                                cell_components
437                                    .last_mut()
438                                    .expect("E: unreachable")
439                                    .push(*vertex_id as usize);
440                                take_next -= 1;
441                            }
442                        }
443                        assert_eq!(num_cells as usize, cell_components.len());
444
445                        let mut errs =
446                            types
447                                .iter()
448                                .zip(cell_components.iter())
449                                .map(|(cell_type, vids)| match cell_type {
450                                    CellType::Vertex => {
451                                        if_predicate_return_err!(
452                                            vids.len() != 1,
453                                            BuilderError::BadVtkData(
454                                                "`Vertex` with incorrect # of vertices (!=1)"
455                                            )
456                                        );
457                                        // silent ignore
458                                        Ok(())
459                                    }
460                                    CellType::PolyVertex => Err(BuilderError::UnsupportedVtkData(
461                                        "`PolyVertex` cell type",
462                                    )),
463                                    CellType::Line => {
464                                        if_predicate_return_err!(
465                                            vids.len() != 2,
466                                            BuilderError::BadVtkData(
467                                                "`Line` with incorrect # of vertices (!=2)"
468                                            )
469                                        );
470                                        // silent ignore
471                                        Ok(())
472                                    }
473                                    CellType::PolyLine => Err(BuilderError::UnsupportedVtkData(
474                                        "`PolyLine` cell type",
475                                    )),
476                                    CellType::Triangle => {
477                                        // check validity
478                                        if_predicate_return_err!(
479                                            vids.len() != 3,
480                                            BuilderError::BadVtkData(
481                                                "`Triangle` with incorrect # of vertices (!=3)"
482                                            )
483                                        );
484                                        // build the triangle
485                                        let d0 = cmap.allocate_used_darts(3);
486                                        let (d1, d2) = (d0 + 1, d0 + 2);
487                                        cmap.force_write_vertex(
488                                            d0 as VertexIdType,
489                                            vertices[vids[0]],
490                                        );
491                                        cmap.force_write_vertex(
492                                            d1 as VertexIdType,
493                                            vertices[vids[1]],
494                                        );
495                                        cmap.force_write_vertex(
496                                            d2 as VertexIdType,
497                                            vertices[vids[2]],
498                                        );
499                                        cmap.force_link::<1>(d0, d1).unwrap(); // edge d0 links vertices vids[0] & vids[1]
500                                        cmap.force_link::<1>(d1, d2).unwrap(); // edge d1 links vertices vids[1] & vids[2]
501                                        cmap.force_link::<1>(d2, d0).unwrap(); // edge d2 links vertices vids[2] & vids[0]
502                                        // record a trace of the built cell for future 2-sew
503                                        sew_buffer.insert((vids[0], vids[1]), d0);
504                                        sew_buffer.insert((vids[1], vids[2]), d1);
505                                        sew_buffer.insert((vids[2], vids[0]), d2);
506                                        Ok(())
507                                    }
508                                    CellType::TriangleStrip => {
509                                        Err(BuilderError::UnsupportedVtkData(
510                                            "`TriangleStrip` cell type",
511                                        ))
512                                    }
513                                    CellType::Polygon => {
514                                        let n_vertices = vids.len();
515                                        let d0 = cmap.allocate_used_darts(n_vertices);
516                                        (0..n_vertices).for_each(|i| {
517                                            let di = d0 + i as DartIdType;
518                                            let dip1 =
519                                                if i == n_vertices - 1 { d0 } else { di + 1 };
520                                            cmap.force_write_vertex(
521                                                di as VertexIdType,
522                                                vertices[vids[i]],
523                                            );
524                                            cmap.force_link::<1>(di, dip1).unwrap();
525                                            sew_buffer
526                                                .insert((vids[i], vids[(i + 1) % n_vertices]), di);
527                                        });
528                                        Ok(())
529                                    }
530                                    CellType::Pixel => {
531                                        Err(BuilderError::UnsupportedVtkData("`Pixel` cell type"))
532                                    }
533                                    CellType::Quad => {
534                                        if_predicate_return_err!(
535                                            vids.len() != 4,
536                                            BuilderError::BadVtkData(
537                                                "`Quad` with incorrect # of vertices (!=4)"
538                                            )
539                                        );
540                                        // build the quad
541                                        let d0 = cmap.allocate_used_darts(4);
542                                        let (d1, d2, d3) = (d0 + 1, d0 + 2, d0 + 3);
543                                        cmap.force_write_vertex(
544                                            d0 as VertexIdType,
545                                            vertices[vids[0]],
546                                        );
547                                        cmap.force_write_vertex(
548                                            d1 as VertexIdType,
549                                            vertices[vids[1]],
550                                        );
551                                        cmap.force_write_vertex(
552                                            d2 as VertexIdType,
553                                            vertices[vids[2]],
554                                        );
555                                        cmap.force_write_vertex(
556                                            d3 as VertexIdType,
557                                            vertices[vids[3]],
558                                        );
559                                        cmap.force_link::<1>(d0, d1).unwrap(); // edge d0 links vertices vids[0] & vids[1]
560                                        cmap.force_link::<1>(d1, d2).unwrap(); // edge d1 links vertices vids[1] & vids[2]
561                                        cmap.force_link::<1>(d2, d3).unwrap(); // edge d2 links vertices vids[2] & vids[3]
562                                        cmap.force_link::<1>(d3, d0).unwrap(); // edge d3 links vertices vids[3] & vids[0]
563                                        // record a trace of the built cell for future 2-sew
564                                        sew_buffer.insert((vids[0], vids[1]), d0);
565                                        sew_buffer.insert((vids[1], vids[2]), d1);
566                                        sew_buffer.insert((vids[2], vids[3]), d2);
567                                        sew_buffer.insert((vids[3], vids[0]), d3);
568                                        Ok(())
569                                    }
570                                    _ => Err(BuilderError::UnsupportedVtkData(
571                                        "CellType not supported in 2-maps",
572                                    )),
573                                });
574                        if let Some(is_err) = errs.find(Result::is_err) {
575                            return Err(is_err.unwrap_err()); // unwrap & wrap because type inference is clunky
576                        }
577                    }
578                    VertexNumbers::XML { .. } => {
579                        return Err(BuilderError::UnsupportedVtkData("XML format"));
580                    }
581                }
582                Ok(())
583            });
584            // return the first error if there is one
585            if let Some(is_err) = tmp.find(Result::is_err) {
586                return Err(is_err.unwrap_err()); // unwrap & wrap because type inference is clunky
587            }
588        }
589    }
590    while let Some(((id0, id1), dart_id0)) = sew_buffer.pop_first() {
591        if let Some(dart_id1) = sew_buffer.remove(&(id1, id0)) {
592            cmap.force_sew::<2>(dart_id0, dart_id1).unwrap();
593        }
594    }
595    Ok(cmap)
596}
