1//! 1D sew implementations

3use crate::{
4    attributes::UnknownAttributeStorage,
5    cmap::{CMap3, DartIdType, NULL_DART_ID, OrbitPolicy, SewError},
6    geometry::CoordsFloat,
7    stm::{Transaction, TransactionClosureResult, try_or_coerce},
8};
9
10#[doc(hidden)]
11/// **1-(un)sews internals)**
12impl<T: CoordsFloat> CMap3<T> {
13    /// 1-sew transactional operation.
14    pub(crate) fn one_sew(
15        &self,
16        t: &mut Transaction,
17        ld: DartIdType,
18        rd: DartIdType,
19    ) -> TransactionClosureResult<(), SewError> {
20        // the main difference with 2D implementation is the beta 3 image check
21        // if both darts have a b3 image, then we need to 1-link b3(rd) to b3(ld) as well
22        // this is handled by `one_link`, but we need to merge old vertex data
23        let b3ld = self.beta_tx::<3>(t, ld)?.max(self.beta_tx::<2>(t, ld)?);
24        let vid_l_old = self.vertex_id_tx(t, b3ld)?;
25        let vid_r_old = self.vertex_id_tx(t, rd)?;
26
27        try_or_coerce!(self.one_link(t, ld, rd), SewError);
28
29        if b3ld != NULL_DART_ID {
30            let new_vid = vid_r_old.min(vid_l_old);
31            try_or_coerce!(
32                self.vertices.merge(t, new_vid, vid_l_old, vid_r_old),
33                SewError
34            );
35            try_or_coerce!(
36                self.attributes.merge_attributes(
37                    t,
38                    OrbitPolicy::Vertex,
39                    new_vid,
40                    vid_l_old,
41                    vid_r_old
42                ),
43                SewError
44            );
45        }
46        Ok(())
47    }
48
49    /// 1-unsew transactional operation.
50    pub(crate) fn one_unsew(
51        &self,
52        t: &mut Transaction,
53        ld: DartIdType,
54    ) -> TransactionClosureResult<(), SewError> {
55        let rd = self.beta_tx::<1>(t, ld)?;
56        let b3ld = self.beta_tx::<3>(t, ld)?.max(self.beta_tx::<2>(t, ld)?);
57
58        try_or_coerce!(self.one_unlink(t, ld), SewError);
59
60        let vid_l_new = self.vertex_id_tx(t, b3ld)?;
61        let vid_r_new = self.vertex_id_tx(t, rd)?;
62        // perf: branch miss vs redundancy
63        if b3ld != NULL_DART_ID && vid_l_new != vid_r_new {
64            try_or_coerce!(
65                self.vertices
66                    .split(t, vid_l_new, vid_r_new, vid_l_new.min(vid_r_new)),
67                SewError
68            );
69            try_or_coerce!(
70                self.attributes.split_attributes(
71                    t,
72                    OrbitPolicy::Vertex,
73                    vid_l_new,
74                    vid_r_new,
75                    vid_l_new.min(vid_r_new),
76                ),
77                SewError
78            );
79        }
80        Ok(())
81    }
82}
