10/// **3-(un)sews internals**
11impl<T: CoordsFloat> CMap3<T> {
12    /// 3-sew operation.
13    #[allow(clippy::too_many_lines)]
14    pub(crate) fn three_sew(
15        &self,
16        t: &mut Transaction,
17        ld: DartIdType,
18        rd: DartIdType,
19    ) -> TransactionClosureResult<(), SewError> {
20        // using these custom orbits, I can get both dart of all sides, directly ordered
21        // for the merges
22        let l_face = self
23            .orbit(OrbitPolicy::Custom(&[1, 0]), ld)
24            .min()
25            .expect("E: unreachable");
26        let r_face = self
27            .orbit(OrbitPolicy::Custom(&[0, 1]), rd)
28            .min()
29            .expect("E: unreachable");
30        let mut edges: Vec<(EdgeIdType, EdgeIdType)> = Vec::with_capacity(10);
31        let mut vertices: Vec<(VertexIdType, VertexIdType)> = Vec::with_capacity(10);
32
33        // read edge + vertex on the b1ld side. if b0ld == NULL, we need to read the left vertex
34        for (l, r) in self
35            .orbit(OrbitPolicy::Custom(&[1, 0]), ld)
36            .zip(self.orbit(OrbitPolicy::Custom(&[0, 1]), rd))
37        {
38            edges.push((self.edge_id_tx(t, l)?, self.edge_id_tx(t, r)?));
39            let b1l = self.beta_tx::<1>(t, l)?;
40            let b2l = self.beta_tx::<2>(t, l)?;
41            // this monster statement is necessary to handle open faces
42            vertices.push((
43                self.vertex_id_tx(t, if b1l == NULL_DART_ID { b2l } else { b1l })?,
44                self.vertex_id_tx(t, r)?,
45            ));
46            // one more for good measures (aka open faces)
47            if self.beta_tx::<0>(t, l)? == NULL_DART_ID {
48                let b1r = self.beta_tx::<1>(t, r)?;
49                let b2r = self.beta_tx::<2>(t, r)?;
50                vertices.push((
51                    self.vertex_id_tx(t, l)?,
52                    self.vertex_id_tx(t, if b1r == NULL_DART_ID { b2r } else { b1r })?,
53                ));
54            }
55        }
56
57        // FIXME: we only check orientation of the arg darts
58        // ideally, we want to check every sewn pair
59        {
60            let (l, r) = (ld, rd);
61            let (b1l, b2l, b1r, b2r) = (
62                self.beta_tx::<1>(t, l)?,
63                self.beta_tx::<2>(t, l)?,
64                self.beta_tx::<1>(t, r)?,
65                self.beta_tx::<2>(t, r)?,
66            );
67            let (vid_l, vid_r, vid_b1l, vid_b1r) = (
68                self.vertex_id_tx(t, l)?,
69                self.vertex_id_tx(t, r)?,
70                self.vertex_id_tx(t, if b1l == NULL_DART_ID { b2l } else { b1l })?,
71                self.vertex_id_tx(t, if b1r == NULL_DART_ID { b2r } else { b1r })?,
72            );
73
74            if let (
75                // (lhs/b1rhs) vertices
76                Some(l_vertex),
77                Some(b1r_vertex),
78                // (b1lhs/rhs) vertices
79                Some(b1l_vertex),
80                Some(r_vertex),
81            ) = (
82                // (lhs/b1rhs)
83                self.vertices.read(t, vid_l)?,
84                self.vertices.read(t, vid_b1r)?,
85                // (b1lhs/rhs)
86                self.vertices.read(t, vid_b1l)?,
87                self.vertices.read(t, vid_r)?,
88            ) {
89                let lhs_vector = b1l_vertex - l_vertex;
90                let rhs_vector = b1r_vertex - r_vertex;
91                // dot product should be negative if the two darts have opposite direction
92                // we could also put restriction on the angle made by the two darts to prevent
93                // drastic deformation
94                if lhs_vector.dot(&rhs_vector) >= T::zero() {
95                    abort(SewError::BadGeometry(3, ld, rd))?;
96                }
97            }
98        }
99
100        // (*): these branch corresponds to incomplete merges (at best),
101        //      or incorrect structure (at worst). that's not a problem
102        //      because `three_link` will detect inconsistencies
103        try_or_coerce!(self.three_link(t, ld, rd), SewError);
104
105        // merge face, edge, vertex attributes
106        try_or_coerce!(
107            self.attributes.merge_attributes(
108                t,
109                OrbitPolicy::Face,
110                l_face.min(r_face),
111                l_face,
112                r_face
113            ),
114            SewError
115        );
116
117        for (eid_l, eid_r) in edges.into_iter().filter(|&(eid_l, eid_r)| {
118            eid_l != eid_r && eid_l != NULL_DART_ID && eid_r != NULL_DART_ID
119        }) {
120            try_or_coerce!(
121                self.attributes.merge_attributes(
122                    t,
123                    OrbitPolicy::Edge,
124                    eid_l.min(eid_r),
125                    eid_l,
126                    eid_r
127                ),
128                SewError
129            );
130        }
131        for (vid_l, vid_r) in vertices.into_iter().filter(|&(vid_l, vid_r)| {
132            vid_l != vid_r && vid_l != NULL_DART_ID && vid_r != NULL_DART_ID
133        }) {
134            try_or_coerce!(
135                self.vertices.merge(t, vid_l.min(vid_r), vid_l, vid_r),
136                SewError
137            );
138            try_or_coerce!(
139                self.attributes.merge_attributes(
140                    t,
141                    OrbitPolicy::Vertex,
142                    vid_l.min(vid_r),
143                    vid_l,
144                    vid_r
145                ),
146                SewError
147            );
148        }
149
150        Ok(())
151    }

153    /// 3-unsew operation.
154    pub(crate) fn three_unsew(
155        &self,
156        t: &mut Transaction,
157        ld: DartIdType,
158    ) -> TransactionClosureResult<(), SewError> {
159        let rd = self.beta_tx::<3>(t, ld)?;
160
161        try_or_coerce!(self.unlink::<3>(t, ld), SewError);
162
163        // faces
164        let l_face = self
165            .orbit(OrbitPolicy::Custom(&[1, 0]), ld)
166            .min()
167            .expect("E: unreachable");
168        let r_face = self
169            .orbit(OrbitPolicy::Custom(&[0, 1]), rd)
170            .min()
171            .expect("E: unreachable");
172        try_or_coerce!(
173            self.attributes.split_attributes(
174                t,
175                OrbitPolicy::Face,
176                l_face,
177                r_face,
178                l_face.min(r_face)
179            ),
180            SewError
181        );
182
183        for (l, r) in self
184            .orbit(OrbitPolicy::Custom(&[1, 0]), ld)
185            .zip(self.orbit(OrbitPolicy::Custom(&[0, 1]), rd))
186        {
187            // edge
188            let (eid_l, eid_r) = (self.edge_id_tx(t, l)?, self.edge_id_tx(t, r)?);
189            try_or_coerce!(
190                self.attributes.split_attributes(
191                    t,
192                    OrbitPolicy::Edge,
193                    eid_l,
194                    eid_r,
195                    eid_l.min(eid_r)
196                ),
197                SewError
198            );
199
200            // vertices
201            let b1l = self.beta_tx::<1>(t, l)?;
202            let b2l = self.beta_tx::<2>(t, l)?;
203            let (vid_l, vid_r) = (
204                self.vertex_id_tx(t, if b1l == NULL_DART_ID { b2l } else { b1l })?,
205                self.vertex_id_tx(t, r)?,
206            );
207            try_or_coerce!(
208                self.vertices.split(t, vid_l, vid_r, vid_l.min(vid_r)),
209                SewError
210            );
211            try_or_coerce!(
212                self.attributes.split_attributes(
213                    t,
214                    OrbitPolicy::Vertex,
215                    vid_l,
216                    vid_r,
217                    vid_l.min(vid_r)
218                ),
219                SewError
220            );
221            if self.beta_tx::<0>(t, l)? == NULL_DART_ID {
222                let b1r = self.beta_tx::<1>(t, r)?;
223                let b2r = self.beta_tx::<2>(t, r)?;
224                let (lvid_l, lvid_r) = (
225                    self.vertex_id_tx(t, l)?,
226                    self.vertex_id_tx(t, if b1r == NULL_DART_ID { b2r } else { b1r })?,
227                );
228                try_or_coerce!(
229                    self.vertices.split(t, lvid_l, lvid_r, lvid_l.min(lvid_r)),
230                    SewError
231                );
232                try_or_coerce!(
233                    self.attributes.split_attributes(
234                        t,
235                        OrbitPolicy::Vertex,
236                        lvid_l,
237                        lvid_r,
238                        lvid_l.min(lvid_r),
239                    ),
240                    SewError
241                );
242            }
243        }
244        Ok(())
245    }
