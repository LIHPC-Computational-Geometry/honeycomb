14    /// 3-sew operation.
15    #[allow(clippy::too_many_lines)]
16    pub(crate) fn three_sew_tx(
17        &self,
18        t: &mut Transaction,
19        ld: DartIdType,
20        rd: DartIdType,
21    ) -> TransactionClosureResult<(), SewError> {
22        // build orbits of each side of the future face
23        // the traversal is equivalent to a beta1/beta0 and beta0/beta1 orbit on respective sides,
24        // we do it manually to detect asymmetry
25        let mut l_side = Vec::with_capacity(10);
26        let mut r_side = Vec::with_capacity(10);
27        l_side.push(ld);
28        r_side.push(rd);
29        let (mut l, mut r) = (self.beta_tx::<1>(t, ld)?, self.beta_tx::<0>(t, rd)?);
30        let mut open = false;
31        // while we haven't looped, or reached an end
32        while l != ld && l != NULL_DART_ID {
33            if r == NULL_DART_ID {
34                // (*)
35                abort(SewError::FailedLink(LinkError::AsymmetricalFaces(ld, rd)))?;
36            }
37            l_side.push(l);
38            r_side.push(r);
39            (l, r) = (self.beta_tx::<1>(t, l)?, self.beta_tx::<0>(t, r)?);
40        }
41        if l == NULL_DART_ID {
42            open = true;
43            // the face was open, so we need to cover the other direction
44            // for meshes, we should be working on complete faces at all times,
45            // so branch prediction will hopefully save use
46            if r != NULL_DART_ID {
47                // if we land on NULL on one side, the other side should be NULL as well
48                // if that is not the case, it means (either):
49                //
50                // - we're trying to sew open faces with a different number of darts
51                // - we're trying to sew open faces that are offset by one (or more) dart(s)
52                //
53                // in both case, this is way too clunky to be considered valid
54                abort(SewError::FailedLink(LinkError::AsymmetricalFaces(ld, rd)))?;
55            }
56            (l, r) = (self.beta_tx::<0>(t, ld)?, self.beta_tx::<1>(t, rd)?);
57            while l != NULL_DART_ID {
58                if r == NULL_DART_ID {
59                    // if we land on NULL on one side, the other side should be NULL as well
60                    abort(SewError::FailedLink(LinkError::AsymmetricalFaces(ld, rd)))?;
61                }
62                l_side.push(l);
63                r_side.push(r);
64                (l, r) = (self.beta_tx::<0>(t, l)?, self.beta_tx::<1>(t, r)?);
65            }
66        }
67
68        let l_face = l_side.iter().min().copied().expect("E: unreachable");
69        let r_face = r_side.iter().min().copied().expect("E: unreachable");
70        let mut edges: Vec<(EdgeIdType, EdgeIdType)> = Vec::with_capacity(10);
71        let mut vertices: Vec<(VertexIdType, VertexIdType)> = Vec::with_capacity(10);
72
73        // read edge + vertex on the b1ld side. if b0ld == NULL, we need to read the left vertex
74        for (&l, &r) in l_side.iter().zip(r_side.iter()) {
75            edges.push((self.edge_id_tx(t, l)?, self.edge_id_tx(t, r)?));
76            let (b1l, b2l) = (self.beta_tx::<1>(t, l)?, self.beta_tx::<2>(t, l)?);
77            // this monster statement is necessary to handle open faces
78            vertices.push((
79                self.vertex_id_tx(t, b1l.max(b2l))?,
80                self.vertex_id_tx(t, r)?,
81            ));
82            // one more for good measures (aka open faces)
83            if self.beta_tx::<0>(t, l)? == NULL_DART_ID {
84                let (b1r, b2r) = (self.beta_tx::<1>(t, r)?, self.beta_tx::<2>(t, r)?);
85                vertices.push((
86                    self.vertex_id_tx(t, l)?,
87                    self.vertex_id_tx(t, b1r.max(b2r))?,
88                ));
89            }
90        }
91
92        // FIXME: we only check orientation of the arg darts
93        // ideally, we want to check every sewn pair
94        {
95            let (vid_l, vid_r, vid_b1l, vid_b1r) = if open {
96                let (l, r) = (ld, rd);
97                let (b1l, b2l, b1r, b2r) = (
98                    self.beta_tx::<1>(t, l)?,
99                    self.beta_tx::<2>(t, l)?,
100                    self.beta_tx::<1>(t, r)?,
101                    self.beta_tx::<2>(t, r)?,
102                );
103                (
104                    self.vertex_id_tx(t, l)?,
105                    self.vertex_id_tx(t, r)?,
106                    self.vertex_id_tx(t, b1l.max(b2l))?,
107                    self.vertex_id_tx(t, b1r.max(b2r))?,
108                )
109            } else {
110                let (vid_b1l, vid_r) = vertices[0];
111                let &(vid_l, vid_b1r) = vertices.last().unwrap();
112                (vid_l, vid_r, vid_b1l, vid_b1r)
113            };
114
115            if let (
116                // (lhs/b1rhs) vertices
117                Some(l_vertex),
118                Some(b1r_vertex),
119                // (b1lhs/rhs) vertices
120                Some(b1l_vertex),
121                Some(r_vertex),
122            ) = (
123                // (lhs/b1rhs)
124                self.vertices.read(t, vid_l)?,
125                self.vertices.read(t, vid_b1r)?,
126                // (b1lhs/rhs)
127                self.vertices.read(t, vid_b1l)?,
128                self.vertices.read(t, vid_r)?,
129            ) {
130                let lhs_vector = b1l_vertex - l_vertex;
131                let rhs_vector = b1r_vertex - r_vertex;
132                // dot product should be negative if the two darts have opposite direction
133                // we could also put restriction on the angle made by the two darts to prevent
134                // drastic deformation
135                if lhs_vector.dot(&rhs_vector) >= T::zero() {
136                    abort(SewError::BadGeometry(3, ld, rd))?;
137                }
138            }
139        }
140
141        // topology update
142        for (l, r) in l_side.into_iter().zip(r_side) {
143            try_or_coerce!(self.betas.three_link_core(t, l, r), SewError);
144        }
145
146        // merge face, edge, vertex attributes
147        try_or_coerce!(
148            self.attributes.merge_attributes(
149                t,
150                OrbitPolicy::Face,
151                l_face.min(r_face),
152                l_face,
153                r_face
154            ),
155            SewError
156        );
157        for (eid_l, eid_r) in edges.into_iter().filter(|&(eid_l, eid_r)| {
158            eid_l != eid_r && eid_l != NULL_DART_ID && eid_r != NULL_DART_ID
159        }) {
160            try_or_coerce!(
161                self.attributes.merge_attributes(
162                    t,
163                    OrbitPolicy::Edge,
164                    eid_l.min(eid_r),
165                    eid_l,
166                    eid_r
167                ),
168                SewError
169            );
170        }
171        for (vid_l, vid_r) in vertices.into_iter().filter(|&(vid_l, vid_r)| {
172            vid_l != vid_r && vid_l != NULL_DART_ID && vid_r != NULL_DART_ID
173        }) {
174            try_or_coerce!(
175                self.vertices.merge(t, vid_l.min(vid_r), vid_l, vid_r),
176                SewError
177            );
178            try_or_coerce!(
179                self.attributes.merge_attributes(
180                    t,
181                    OrbitPolicy::Vertex,
182                    vid_l.min(vid_r),
183                    vid_l,
184                    vid_r
185                ),
186                SewError
187            );
188        }
189
190        Ok(())
191    }

194    /// 3-unsew operation.
195    pub(crate) fn three_unsew_tx(
196        &self,
197        t: &mut Transaction,
198        ld: DartIdType,
199    ) -> TransactionClosureResult<DartIdType, SewError> {
200        let rd = try_or_coerce!(self.betas.three_unlink_core(t, ld), SewError);
201        let mut l_side = Vec::with_capacity(10);
202        let mut r_side = Vec::with_capacity(10);
203        l_side.push(ld);
204        r_side.push(rd);
205
206        let (mut l, mut r) = (self.beta_tx::<1>(t, ld)?, self.beta_tx::<0>(t, rd)?);
207        // while we haven't completed the loop, or reached an end
208        while l != ld && l != NULL_DART_ID {
209            if l != self.beta_tx::<3>(t, r)? {
210                // (*); FIXME: add dedicated err ~LinkError::DivergentStructures ?
211                abort(SewError::FailedLink(LinkError::AsymmetricalFaces(ld, rd)))?;
212            }
213            try_or_coerce!(self.betas.three_unlink_core(t, l), SewError);
214            l_side.push(l);
215            r_side.push(r);
216            (l, r) = (self.beta_tx::<1>(t, l)?, self.beta_tx::<0>(t, r)?);
217        }
218        // the face was open, so we need to cover the other direction
219        // for meshes, we should be working on complete faces at all times,
220        // so branch prediction will hopefully save use
221        if l == NULL_DART_ID {
222            if r != NULL_DART_ID {
223                // if we land on NULL on one side, the other side should be NULL as well
224                abort(SewError::FailedLink(LinkError::AsymmetricalFaces(ld, rd)))?;
225            }
226            (l, r) = (self.beta_tx::<0>(t, ld)?, self.beta_tx::<1>(t, rd)?);
227            while l != NULL_DART_ID {
228                if l != self.beta_tx::<3>(t, r)? {
229                    // this can be changed, but the idea here is to ensure we're unlinking
230                    // the expected construct
231                    // FIXME: add dedicated err ~LinkError::DivergentStructures ?
232                    abort(SewError::FailedLink(LinkError::AsymmetricalFaces(ld, rd)))?;
233                }
234                assert_eq!(l, self.beta_tx::<3>(t, r)?); // (*)
235                try_or_coerce!(self.betas.three_unlink_core(t, l), SewError);
236                l_side.push(l);
237                r_side.push(r);
238                (l, r) = (self.beta_tx::<0>(t, l)?, self.beta_tx::<1>(t, r)?);
239            }
240        }
241
242        // faces
243        let l_face = l_side.iter().min().copied().expect("E: unreachable");
244        let r_face = r_side.iter().min().copied().expect("E: unreachable");
245        try_or_coerce!(
246            self.attributes.split_attributes(
247                t,
248                OrbitPolicy::Face,
249                l_face,
250                r_face,
251                l_face.min(r_face)
252            ),
253            SewError
254        );
255
256        for (l, r) in l_side.into_iter().zip(r_side.into_iter()) {
257            // edge
258            let (eid_l, eid_r) = (self.edge_id_tx(t, l)?, self.edge_id_tx(t, r)?);
259            if eid_l != eid_r {
260                try_or_coerce!(
261                    self.attributes.split_attributes(
262                        t,
263                        OrbitPolicy::Edge,
264                        eid_l,
265                        eid_r,
266                        eid_l.min(eid_r)
267                    ),
268                    SewError
269                );
270            }
271
272            // vertices
273            let (b1l, b2l) = (self.beta_tx::<1>(t, l)?, self.beta_tx::<2>(t, l)?);
274            let (vid_l, vid_r) = (
275                self.vertex_id_tx(t, b1l.max(b2l))?,
276                self.vertex_id_tx(t, r)?,
277            );
278            if vid_l != vid_r {
279                try_or_coerce!(
280                    self.vertices.split(t, vid_l, vid_r, vid_l.min(vid_r)),
281                    SewError
282                );
283                try_or_coerce!(
284                    self.attributes.split_attributes(
285                        t,
286                        OrbitPolicy::Vertex,
287                        vid_l,
288                        vid_r,
289                        vid_l.min(vid_r)
290                    ),
291                    SewError
292                );
293            }
294            if self.beta_tx::<0>(t, l)? == NULL_DART_ID {
295                let (b1r, b2r) = (self.beta_tx::<1>(t, r)?, self.beta_tx::<2>(t, r)?);
296                let (lvid_l, lvid_r) = (
297                    self.vertex_id_tx(t, l)?,
298                    self.vertex_id_tx(t, b1r.max(b2r))?,
299                );
300                if lvid_l != lvid_r {
301                    try_or_coerce!(
302                        self.vertices.split(t, lvid_l, lvid_r, lvid_l.min(lvid_r)),
303                        SewError
304                    );
305                    try_or_coerce!(
306                        self.attributes.split_attributes(
307                            t,
308                            OrbitPolicy::Vertex,
309                            lvid_l,
310                            lvid_r,
311                            lvid_l.min(lvid_r),
312                        ),
313                        SewError
314                    );
315                }
316            }
317        }
318        Ok(rd)
319    }
