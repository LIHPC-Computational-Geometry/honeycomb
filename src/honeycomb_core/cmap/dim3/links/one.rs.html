1//! 1D link implementations
2
3use crate::cmap::{CMap3, DartIdType, LinkError, NULL_DART_ID};
4use crate::geometry::CoordsFloat;
5use crate::stm::{Transaction, TransactionClosureResult, abort, atomically_with_err};
6
7/// 1-links
8impl<T: CoordsFloat> CMap3<T> {
9    /// 1-link operation.
10    pub(crate) fn one_link(
11        &self,
12        t: &mut Transaction,
13        ld: DartIdType,
14        rd: DartIdType,
15    ) -> TransactionClosureResult<(), LinkError> {
16        self.betas.one_link_core(t, ld, rd)?;
17        let (b3_ld, b3_rd) = (self.beta_tx::<3>(t, ld)?, self.beta_tx::<3>(t, rd)?);
18        if b3_ld != NULL_DART_ID && b3_rd != NULL_DART_ID {
19            self.betas.one_link_core(t, b3_rd, b3_ld)?;
20        }
21        Ok(())
22    }
23
24    /// 1-link operation.
25    ///
26    /// This variant is equivalent to `one_link`, but internally uses a transaction that will be
27    /// retried until validated.
28    pub(crate) fn force_one_link(&self, ld: DartIdType, rd: DartIdType) -> Result<(), LinkError> {
29        atomically_with_err(|t| self.one_link(t, ld, rd))
30    }
31}
32
33/// 1-unlinks
34impl<T: CoordsFloat> CMap3<T> {
35    /// 1-unlink operation.
36    pub(crate) fn one_unlink(
37        &self,
38        t: &mut Transaction,
39        ld: DartIdType,
40    ) -> TransactionClosureResult<(), LinkError> {
41        let rd = self.beta_tx::<1>(t, ld)?;
42        self.betas.one_unlink_core(t, ld)?;
43        let (b3_ld, b3_rd) = (self.beta_tx::<3>(t, ld)?, self.beta_tx::<3>(t, rd)?);
44        if b3_ld != NULL_DART_ID && b3_rd != NULL_DART_ID {
45            if self.beta_tx::<1>(t, b3_rd)? != b3_ld {
46                // FIXME: add dedicated variant ~LinkError::DivergentStructures ?
47                abort(LinkError::AsymmetricalFaces(ld, rd))?;
48            }
49            self.betas.one_unlink_core(t, b3_rd)?;
50        }
51        Ok(())
52    }
53
54    /// 1-unlink operation.
55    ///
56    /// This variant is equivalent to `one_unlink`, but internally uses a transaction that will be
57    /// retried until validated.
58    pub(crate) fn force_one_unlink(&self, ld: DartIdType) -> Result<(), LinkError> {
59        atomically_with_err(|t| self.one_unlink(t, ld))
60    }
61}
