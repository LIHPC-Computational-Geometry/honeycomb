7/// 3-links
8impl<T: CoordsFloat> CMap3<T> {
9    /// 3-link operation.
10    pub(crate) fn three_link(
11        &self,
12        t: &mut Transaction,
13        ld: DartIdType,
14        rd: DartIdType,
15    ) -> TransactionClosureResult<(), LinkError> {
16        self.betas.three_link_core(t, ld, rd)?;
17        let (mut lside, mut rside) = (self.beta_tx::<1>(t, ld)?, self.beta_tx::<0>(t, rd)?);
18        // while we haven't completed the loop, or reached an end
19        while lside != ld && lside != NULL_DART_ID {
20            if rside == NULL_DART_ID {
21                // (*)
22                abort(LinkError::AsymmetricalFaces(ld, rd))?;
23            }
24            self.betas.three_link_core(t, lside, rside)?;
25            (lside, rside) = (self.beta_tx::<1>(t, lside)?, self.beta_tx::<0>(t, rside)?);
26        }
27        // the face was open, so we need to cover the other direction
28        // for meshes, we should be working on complete faces at all times,
29        // so branch prediction will hopefully save use
30        if lside == NULL_DART_ID {
31            if rside != NULL_DART_ID {
32                // (*)
33                abort(LinkError::AsymmetricalFaces(ld, rd))?;
34            }
35            (lside, rside) = (self.beta_tx::<0>(t, ld)?, self.beta_tx::<1>(t, rd)?);
36            while lside != NULL_DART_ID {
37                if rside == NULL_DART_ID {
38                    // (*)
39                    abort(LinkError::AsymmetricalFaces(ld, rd))?;
40                }
41                self.betas.three_link_core(t, lside, rside)?;
42                (lside, rside) = (self.beta_tx::<0>(t, lside)?, self.beta_tx::<1>(t, rside)?);
43            }
44        }
45        // (*): if we land on NULL on one side, the other side should be NULL as well
46        //      if that is not the case, it means (either):
47        //      - we're trying to sew open faces with a different number of darts
48        //      - we're trying to sew open faces that are offset by one (or more) dart(s)
49        //      in both case, this is way too clunky to be considered valid
50        Ok(())
51    }
52
53    /// 3-link operation.
54    pub(crate) fn force_three_link(&self, ld: DartIdType, rd: DartIdType) -> Result<(), LinkError> {
55        atomically_with_err(|t| self.three_link(t, ld, rd))
56    }
57}

59/// 3-unlinks
60impl<T: CoordsFloat> CMap3<T> {
61    /// 3-unlink operation.
62    pub(crate) fn three_unlink(
63        &self,
64        t: &mut Transaction,
65        ld: DartIdType,
66    ) -> TransactionClosureResult<(), LinkError> {
67        let rd = self.beta_tx::<3>(t, ld)?;
68
69        self.betas.three_unlink_core(t, ld)?;
70        let (mut lside, mut rside) = (self.beta_tx::<1>(t, ld)?, self.beta_tx::<0>(t, rd)?);
71        // while we haven't completed the loop, or reached an end
72        while lside != ld && lside != NULL_DART_ID {
73            if lside != self.beta_tx::<3>(t, rside)? {
74                // (*); FIXME: add dedicated err ~LinkError::DivergentStructures ?
75                abort(LinkError::AsymmetricalFaces(ld, rd))?;
76            }
77            self.betas.three_unlink_core(t, lside)?;
78            (lside, rside) = (self.beta_tx::<1>(t, lside)?, self.beta_tx::<0>(t, rside)?);
79        }
80        // the face was open, so we need to cover the other direction
81        // for meshes, we should be working on complete faces at all times,
82        // so branch prediction will hopefully save use
83        if lside == NULL_DART_ID {
84            if rside != NULL_DART_ID {
85                // (**)
86                abort(LinkError::AsymmetricalFaces(ld, rd))?;
87            }
88            (lside, rside) = (self.beta_tx::<0>(t, ld)?, self.beta_tx::<1>(t, rd)?);
89            while lside != NULL_DART_ID {
90                if lside != self.beta_tx::<3>(t, rside)? {
91                    // (*); FIXME: add dedicated err ~LinkError::DivergentStructures ?
92                    abort(LinkError::AsymmetricalFaces(ld, rd))?;
93                }
94                assert_eq!(lside, self.beta_tx::<3>(t, rside)?); // (*)
95                self.betas.three_unlink_core(t, lside)?;
96                (lside, rside) = (self.beta_tx::<0>(t, lside)?, self.beta_tx::<1>(t, rside)?);
97            }
98        }
99        // (*) : this can be changed, but the idea here is to ensure we're unlinking the expected
100        //       construct
101        // (**): if we land on NULL on one side, the other side should be NULL as well
102        Ok(())
103    }
104
105    /// 3-unlink operation.
106    pub(crate) fn force_three_unlink(&self, ld: DartIdType) -> Result<(), LinkError> {
107        atomically_with_err(|t| self.three_unlink(t, ld))
108    }
