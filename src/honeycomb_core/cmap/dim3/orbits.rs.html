1//! Orbit implementation
2//!
3//! This module contains all code used to model orbits, a notion defined
4//! along the structure of combinatorial maps.
5
6use std::collections::{HashSet, VecDeque};
7
8use crate::cmap::{CMap3, DartIdType, NULL_DART_ID, OrbitPolicy};
9use crate::geometry::CoordsFloat;
10
11impl<T: CoordsFloat> CMap3<T> {
12    /// Generic orbit implementation.
13    ///
14    /// # Arguments
15    /// - `opolicy: OrbitPolicy` -- Policy used by the orbit for the BFS.
16    /// - `dart_id: DartIdentifier` -- Dart of which the structure will compute the orbit.
17    ///
18    /// # The search algorithm
19    ///
20    /// The search algorithm used to establish the list of dart included in the orbit is a
21    /// [Breadth-First Search algorithm][WIKIBFS]. This means that:
22    ///
23    /// - we look at the images of the current dart through all beta functions,
24    ///   adding those to a queue, before moving on to the next dart.
25    /// - we apply the beta functions in their specified order; This guarantees a consistent and
26    ///   predictable result.
27    ///
28    /// # Performance
29    ///
30    /// Currently, orbits use two dynamically allocated structures for computation: a `VecDeque`,
31    /// and a `HashSet`. There is a possibility to use static thread-local instances to avoid
32    /// ephemeral allocations, but [it would require a guard mechanism][PR].
33    ///
34    /// [PR]: https://github.com/LIHPC-Computational-Geometry/honeycomb/pull/293
35    #[allow(clippy::needless_for_each,clippy::too_many_lines)]
36    #[rustfmt::skip]
37    pub fn orbit(
38        &self,
39        opolicy: OrbitPolicy,
40        dart_id: DartIdType,
41    ) -> impl Iterator<Item = DartIdType> {
42        let mut pending = VecDeque::new();
43        let mut marked: HashSet<DartIdType> = HashSet::new();
44        pending.push_back(dart_id);
45        marked.insert(NULL_DART_ID);
46        marked.insert(dart_id); // we're starting here, so we mark it beforehand
47
48        // FIXME: move the match block out of the iterator
49        std::iter::from_fn(move || {
50            if let Some(d) = pending.pop_front() {
51                // compute the next images
52                match opolicy {
53                    // B3oB2, B1oB3, B1oB2, B3oB0, B2oB0
54                    OrbitPolicy::Vertex => {
55                        [
56                            self.beta::<3>(self.beta::<2>(d)), // b3(b2(d))
57                            self.beta::<1>(self.beta::<3>(d)), // b1(b3(d))
58                            self.beta::<1>(self.beta::<2>(d)), // b1(b2(d))
59                            self.beta::<3>(self.beta::<0>(d)), // b3(b0(d))
60                            self.beta::<2>(self.beta::<0>(d)), // b2(b0(d))
61                        ]
62                        .into_iter()
63                        .for_each(|im| {
64                            if marked.insert(im) {
65                                pending.push_back(im);
66                            }
67                        });
68                    }
69                    // B3oB2, B1oB3, B1oB2
70                    OrbitPolicy::VertexLinear => {
71                        [
72                            self.beta::<3>(self.beta::<2>(d)), // b3(b2(d))
73                            self.beta::<1>(self.beta::<3>(d)), // b1(b3(d))
74                            self.beta::<1>(self.beta::<2>(d)), // b1(b2(d))
75                        ]
76                        .into_iter()
77                        .for_each(|im| {
78                            if marked.insert(im) {
79                                pending.push_back(im);
80                            }
81                        });
82                    }
83                    // B2, B3
84                    OrbitPolicy::Edge => {
85                        [
86                            self.beta::<2>(d),
87                            self.beta::<3>(d),
88                        ]
89                        .into_iter()
90                        .for_each(|im| {
91                            if marked.insert(im) {
92                                pending.push_back(im);
93                            }
94                        });
95                    }
96                    // B1, B0, B3
97                    OrbitPolicy::Face => {
98                        [
99                            self.beta::<1>(d),
100                            self.beta::<0>(d),
101                            self.beta::<3>(d),
102                        ]
103                        .into_iter()
104                        .for_each(|im| {
105                            if marked.insert(im) {
106                                pending.push_back(im);
107                            }
108                        });
109                    }
110                    // B1, B3
111                    OrbitPolicy::FaceLinear => {
112                        [
113                            self.beta::<1>(d),
114                            self.beta::<3>(d),
115                        ]
116                        .into_iter()
117                        .for_each(|im| {
118                            if marked.insert(im) {
119                                pending.push_back(im);
120                            }
121                        });
122                    }
123                    // B1, B0, B2
124                    OrbitPolicy::Volume => {
125                        [
126                            self.beta::<1>(d),
127                            self.beta::<0>(d),
128                            self.beta::<2>(d),
129                        ]
130                        .into_iter()
131                        .for_each(|im| {
132                            if marked.insert(im) {
133                                pending.push_back(im);
134                            }
135                        });
136                    }
137                    // B1, B2
138                    OrbitPolicy::VolumeLinear => {
139                        [
140                            self.beta::<1>(d),
141                            self.beta::<2>(d),
142                        ]
143                        .into_iter()
144                        .for_each(|im| {
145                            if marked.insert(im) {
146                                pending.push_back(im);
147                            }
148                        });
149                    }
150                    OrbitPolicy::Custom(beta_slice) => {
151                        for beta_id in beta_slice {
152                            let image = self.beta_rt(*beta_id, d);
153                            if marked.insert(image) {
154                                pending.push_back(image);
155                            }
156                        }
157                    }
158                }
159
160                return Some(d);
161            }
162            None // queue is empty, we're done
163        })
164    }

166    /// Return the orbit defined by a dart and its `I`-cell.
167    ///
168    /// # Usage
169    ///
170    /// The [`Orbit3`] can be iterated upon to retrieve all dart members of the cell. Note that
171    /// **the dart passed as an argument is included as the first element of the returned orbit**.
172    ///
173    /// # Panics
174    ///
175    /// The method will panic if *I* is not 0, 1, 2, or 3.
176    #[must_use = "unused return value"]
177    pub fn i_cell<const I: u8>(&self, dart_id: DartIdType) -> impl Iterator<Item = DartIdType> {
178        assert!(I < 4);
179        match I {
180            0 => self.orbit(OrbitPolicy::Vertex, dart_id),
181            1 => self.orbit(OrbitPolicy::Edge, dart_id),
182            2 => self.orbit(OrbitPolicy::Face, dart_id),
183            3 => self.orbit(OrbitPolicy::Volume, dart_id),
184            _ => unreachable!(),
185        }
186    }
