11/// Beta functions storage.
12///
13/// `N` is the number of beta function stored, including `B0`. This means that, for example,
14/// a 2-map will have a `BetaFunctions<3>` object field.
15pub struct BetaFunctions<const N: usize>(Vec<[TVar<DartIdType>; N]>);
16
17#[allow(unused)]
18impl<const N: usize> BetaFunctions<N> {
19    #[cfg(not(feature = "par-internals"))]
20    /// Constructor
21    pub fn new(n_darts: usize) -> Self {
22        Self(
23            (0..n_darts)
24                .map(|_| std::array::from_fn(|_| TVar::new(NULL_DART_ID)))
25                .collect(),
26        )
27    }
28
29    #[cfg(feature = "par-internals")]
30    /// Constructor
31    pub fn new(n_darts: usize) -> Self {
32        Self(
33            (0..n_darts)
34                .into_par_iter()
35                .map(|_| std::array::from_fn(|_| TVar::new(NULL_DART_ID)))
36                .collect(),
37        )
38    }
39
40    #[cfg(not(feature = "par-internals"))]
41    /// Extend internal storage capacity
42    pub fn extend(&mut self, len: usize) {
43        self.0
44            .extend((0..len).map(|_| std::array::from_fn(|_| TVar::new(NULL_DART_ID))));
45    }
46
47    #[cfg(feature = "par-internals")]
48    /// Extend internal storage capacity
49    pub fn extend(&mut self, len: usize) {
50        self.0.par_extend(
51            (0..len)
52                .into_par_iter()
53                .map(|_| std::array::from_fn(|_| TVar::new(NULL_DART_ID))),
54        );
55    }
56
57    /// Return internal storage capacity
58    pub fn capacity(&self) -> usize {
59        self.0.capacity()
60    }
61}

63impl<const N: usize> Index<(u8, DartIdType)> for BetaFunctions<N> {
64    type Output = TVar<DartIdType>;
65
66    fn index(&self, (beta_id, dart_id): (u8, DartIdType)) -> &Self::Output {
67        &self.0[dart_id as usize][beta_id as usize]
68    }
69}

71impl<const N: usize> IndexMut<(u8, DartIdType)> for BetaFunctions<N> {
72    fn index_mut(&mut self, (beta_id, dart_id): (u8, DartIdType)) -> &mut Self::Output {
73        &mut self.0[dart_id as usize][beta_id as usize]
74    }
75}

81impl<const N: usize> BetaFunctions<N> {
82    /// 1-link operation.
83    ///
84    ///
85    /// This operation corresponds to linking two darts via the *β<sub>1</sub>* function. Unlike
86    /// its sewing counterpart, this method does not contain any code to update the attributes or
87    /// geometrical data of the affected cell(s). The *β<sub>0</sub>* function is also updated.
88    ///
89    /// # Arguments
90    ///
91    /// - `lhs_dart_id: DartIdentifier` -- ID of the first dart to be linked.
92    /// - `rhs_dart_id: DartIdentifier` -- ID of the second dart to be linked.
93    ///
94    /// # Panics
95    ///
96    /// This method may panic if `lhs_dart_id` isn't 1-free or `rhs_dart_id` isn't 0-free.
97    ///
98    pub fn one_link_core(
99        &self,
100        t: &mut Transaction,
101        lhs_dart_id: DartIdType,
102        rhs_dart_id: DartIdType,
103    ) -> TransactionClosureResult<(), LinkError> {
104        if self[(1, lhs_dart_id)].read(t)? != NULL_DART_ID {
105            return abort(LinkError::NonFreeBase(1, lhs_dart_id, rhs_dart_id));
106        }
107        if self[(0, rhs_dart_id)].read(t)? != NULL_DART_ID {
108            return abort(LinkError::NonFreeImage(0, lhs_dart_id, rhs_dart_id));
109        }
110        // set beta_1(lhs_dart) to rhs_dart
111        self[(1, lhs_dart_id)].write(t, rhs_dart_id)?;
112        // set beta_0(rhs_dart) to lhs_dart
113        self[(0, rhs_dart_id)].write(t, lhs_dart_id)?;
114        Ok(())
115    }
116
117    /// 2-link operation.
118    ///
119    /// This operation corresponds to linking two darts via the *β<sub>2</sub>* function. Unlike
120    /// its sewing counterpart, this method does not contain any code to update the attributes or
121    /// geometrical data of the affected cell(s).
122    ///
123    /// # Arguments
124    ///
125    /// - `lhs_dart_id: DartIdentifier` -- ID of the first dart to be linked.
126    /// - `rhs_dart_id: DartIdentifier` -- ID of the second dart to be linked.
127    ///
128    /// # Panics
129    ///
130    /// This method may panic if one of `lhs_dart_id` or `rhs_dart_id` isn't 2-free.
131    pub fn two_link_core(
132        &self,
133        t: &mut Transaction,
134        lhs_dart_id: DartIdType,
135        rhs_dart_id: DartIdType,
136    ) -> TransactionClosureResult<(), LinkError> {
137        if self[(2, lhs_dart_id)].read(t)? != NULL_DART_ID {
138            return abort(LinkError::NonFreeBase(2, lhs_dart_id, rhs_dart_id));
139        }
140        if self[(2, rhs_dart_id)].read(t)? != NULL_DART_ID {
141            return abort(LinkError::NonFreeImage(2, lhs_dart_id, rhs_dart_id));
142        }
143        // set beta_2(lhs_dart) to rhs_dart
144        self[(2, lhs_dart_id)].write(t, rhs_dart_id)?;
145        // set beta_2(rhs_dart) to lhs_dart
146        self[(2, rhs_dart_id)].write(t, lhs_dart_id)?;
147        Ok(())
148    }
149
150    pub fn three_link_core(
151        &self,
152        t: &mut Transaction,
153        lhs_dart_id: DartIdType,
154        rhs_dart_id: DartIdType,
155    ) -> TransactionClosureResult<(), LinkError> {
156        if self[(3, lhs_dart_id)].read(t)? != NULL_DART_ID {
157            return abort(LinkError::NonFreeBase(3, lhs_dart_id, rhs_dart_id));
158        }
159        if self[(3, rhs_dart_id)].read(t)? != NULL_DART_ID {
160            return abort(LinkError::NonFreeImage(3, lhs_dart_id, rhs_dart_id));
161        }
162        self[(3, lhs_dart_id)].write(t, rhs_dart_id)?;
163        self[(3, rhs_dart_id)].write(t, lhs_dart_id)?;
164        Ok(())
165    }
166
167    /// 1-unlink operation.
168    ///
169    /// This operation corresponds to unlinking two darts that are linked via the *β<sub>1</sub>*
170    /// function. Unlike its sewing counterpart, this method does not contain any code to update
171    /// the attributes or geometrical data of the affected cell(s). The *β<sub>0</sub>* function is
172    /// also updated.
173    ///
174    /// # Arguments
175    ///
176    /// - `lhs_dart_id: DartIdentifier` -- ID of the dart to unlink.
177    ///
178    /// # Panics
179    ///
180    /// This method may panic if one of `lhs_dart_id` is already 1-free.
181    pub fn one_unlink_core(
182        &self,
183        t: &mut Transaction,
184        lhs_dart_id: DartIdType,
185    ) -> TransactionClosureResult<(), LinkError> {
186        // set beta_1(lhs_dart) to NullDart
187        let rhs_dart_id = self[(1, lhs_dart_id)].replace(t, NULL_DART_ID)?;
188        if rhs_dart_id == NULL_DART_ID {
189            return abort(LinkError::AlreadyFree(1, lhs_dart_id));
190        }
191        // set beta_0(rhs_dart) to NullDart
192        self[(0, rhs_dart_id)].write(t, NULL_DART_ID)?;
193        Ok(())
194    }
195
196    /// 2-unlink operation.
197    ///
198    /// This operation corresponds to unlinking two darts that are linked via the *β<sub>2</sub>*
199    /// function. Unlike its sewing counterpart, this method does not contain any code to update
200    /// the attributes or geometrical data of the affected cell(s).
201    ///
202    /// # Arguments
203    ///
204    /// - `lhs_dart_id: DartIdentifier` -- ID of the dart to unlink.
205    ///
206    /// # Panics
207    ///
208    /// This method may panic if one of `lhs_dart_id` is already 2-free.
209    pub fn two_unlink_core(
210        &self,
211        t: &mut Transaction,
212        lhs_dart_id: DartIdType,
213    ) -> TransactionClosureResult<(), LinkError> {
214        // set beta_2(dart) to NullDart
215        let rhs_dart_id = self[(2, lhs_dart_id)].replace(t, NULL_DART_ID)?;
216        if rhs_dart_id == NULL_DART_ID {
217            return abort(LinkError::AlreadyFree(2, lhs_dart_id));
218        }
219        // set beta_2(beta_2(dart)) to NullDart
220        self[(2, rhs_dart_id)].write(t, NULL_DART_ID)?;
221        Ok(())
222    }
223
224    pub fn three_unlink_core(
225        &self,
226        t: &mut Transaction,
227        lhs_dart_id: DartIdType,
228    ) -> TransactionClosureResult<(), LinkError> {
229        // set beta_3(lhs_dart) to NullDart
230        let rhs_dart_id = self[(3, lhs_dart_id)].replace(t, NULL_DART_ID)?;
231        if rhs_dart_id == NULL_DART_ID {
232            return abort(LinkError::AlreadyFree(3, lhs_dart_id));
233        }
234        // set beta_3(rhs_dart) to NullDart
235        self[(3, rhs_dart_id)].write(t, NULL_DART_ID)?;
236        Ok(())
237    }
