17// use thread local hashset and queue for orbit traversal of ID comp.
18// not applied to orbit currently bc they are lazily onsumed, and therefore require dedicated
19// instances to be robust
20thread_local! {
21    static AUXILIARIES: RefCell<(VecDeque<DartIdType>, HashSet<DartIdType>)> = RefCell::new((VecDeque::with_capacity(10), HashSet::with_capacity(10)));
22}

24/// **Beta-related methods**
25impl<T: CoordsFloat> CMap2<T> {

28    /// Return  β<sub>`I`</sub>(`dart_id`).
29    ///
30    /// # Panics
31    ///
32    /// The method will panic if `I` is not 0, 1 or 2.
33    #[must_use = "unused return value"]
34    pub fn beta<const I: u8>(&self, dart_id: DartIdType) -> DartIdType {
35        assert!(I < 3);
36        self.betas[(I, dart_id)].read_atomic()
37    }
38
39    /// Return  β<sub>`i`</sub>(`dart_id`).
40    ///
41    /// # Panics
42    ///
43    /// The method will panic if `i` is not 0, 1 or 2.
44    #[must_use = "unused return value"]
45    pub fn beta_rt(&self, i: u8, dart_id: DartIdType) -> DartIdType {
46        assert!(i < 3);
47        match i {
48            0 => self.beta::<0>(dart_id),
49            1 => self.beta::<1>(dart_id),
50            2 => self.beta::<2>(dart_id),
51            _ => unreachable!(),
52        }
53    }
54
55    /// Return  β<sub>`I`</sub>(`dart_id`).
56    ///
57    /// # Errors
58    ///
59    /// This method is meant to be called in a context where the returned `Result` is used to
60    /// validate the transaction passed as argument. Errors should not be processed manually,
61    /// only processed via the `?` operator.
62    ///
63    /// # Panics
64    ///
65    /// The method will panic if `I` is not 0, 1 or 2.
66    pub fn beta_transac<const I: u8>(
67        &self,
68        trans: &mut Transaction,
69        dart_id: DartIdType,
70    ) -> StmClosureResult<DartIdType> {
71        assert!(I < 3);
72        self.betas[(I, dart_id)].read(trans)
73    }
74
75    /// Return  β<sub>`i`</sub>(`dart_id`).
76    ///
77    /// # Errors
78    ///
79    /// This method is meant to be called in a context where the returned `Result` is used to
80    /// validate the transaction passed as argument. Errors should not be processed manually,
81    /// only processed via the `?` operator.
82    ///
83    /// # Panics
84    ///
85    /// The method will panic if `i` is not 0, 1 or 2.
86    pub fn beta_rt_transac(
87        &self,
88        trans: &mut Transaction,
89        i: u8,
90        dart_id: DartIdType,
91    ) -> StmClosureResult<DartIdType> {
92        assert!(i < 3);
93        match i {
94            0 => self.beta_transac::<0>(trans, dart_id),
95            1 => self.beta_transac::<1>(trans, dart_id),
96            2 => self.beta_transac::<2>(trans, dart_id),
97            _ => unreachable!(),
98        }
99    }

101    /// Check if a given dart is `I`-free.
102    ///
103    /// # Return
104    ///
105    /// Return a boolean indicating if the dart is `I`-free, i.e.:
106    /// - `true` if β<sub>`I`</sub>(`dart_id`) = `NULL_DART_ID`,
107    /// - `false` else.
108    ///
109    /// # Panics
110    ///
111    /// The function will panic if *I* is not 0, 1 or 2.
112    ///
113    #[must_use = "unused return value"]
114    pub fn is_i_free<const I: u8>(&self, dart_id: DartIdType) -> bool {
115        self.beta::<I>(dart_id) == NULL_DART_ID
116    }
117
118    /// Check if a given dart is `i`-free, for all `i`.
119    ///
120    /// # Return
121    ///
122    /// Return a boolean indicating if the dart is 0-free, 1-free **and** 2-free.
123    #[must_use = "unused return value"]
124    pub fn is_free(&self, dart_id: DartIdType) -> bool {
125        atomically(|t| self.is_free_transac(t, dart_id))
126    }
127
128    #[allow(clippy::missing_errors_doc)]
129    /// Check if a given dart is `i`-free, for all `i`.
130    ///
131    /// # Return / Errors
132    ///
133    /// Return a boolean indicating if the dart is 0-free, 1-free **and** 2-free.
134    ///
135    /// This method is meant to be called in a context where the returned `Result` is used to
136    /// validate the transaction passed as argument. Errors should not be processed manually,
137    /// only processed via the `?` operator.
138    #[must_use = "unused return value"]
139    pub fn is_free_transac(
140        &self,
141        t: &mut Transaction,
142        dart_id: DartIdType,
143    ) -> StmClosureResult<bool> {
144        Ok(self.beta_transac::<0>(t, dart_id)? == NULL_DART_ID
145            && self.beta_transac::<1>(t, dart_id)? == NULL_DART_ID
146            && self.beta_transac::<2>(t, dart_id)? == NULL_DART_ID)
147    }
148}

150/// **I-cell-related methods**
151impl<T: CoordsFloat> CMap2<T> {
152    /// Compute the ID of the vertex a given dart is part of.
153    ///
154    /// This corresponds to the minimum dart ID among darts composing the 0-cell orbit.
155    #[must_use = "unused return value"]
156    pub fn vertex_id(&self, dart_id: DartIdType) -> VertexIdType {
157        atomically(|trans| self.vertex_id_transac(trans, dart_id))
158    }
159
160    /// Compute the ID of the vertex a given dart is part of.
161    ///
162    /// This corresponds to the minimum dart ID among darts composing the 0-cell orbit.
163    ///
164    /// # Errors
165    ///
166    /// This method is meant to be called in a context where the returned `Result` is used to
167    /// validate the transaction passed as argument. Errors should not be processed manually,
168    /// only processed via the `?` operator.
169    pub fn vertex_id_transac(
170        &self,
171        trans: &mut Transaction,
172        dart_id: DartIdType,
173    ) -> StmClosureResult<VertexIdType> {
174        AUXILIARIES.with(|t| {
175            let (pending, marked) = &mut *t.borrow_mut();
176            // clear from previous computations
177            pending.clear();
178            marked.clear();
179            // initialize
180            pending.push_back(dart_id);
181            marked.insert(NULL_DART_ID); // we don't want to include the null dart in the orbit
182            marked.insert(dart_id); // we're starting here, so we mark it beforehand
183
184            let mut min = dart_id;
185
186            while let Some(d) = pending.pop_front() {
187                // THIS CODE IS ONLY VALID IN 2D
188                let (b2d, b0d) = (
189                    self.beta_transac::<2>(trans, d)?,
190                    self.beta_transac::<0>(trans, d)?,
191                );
192                let image1 = self.beta_transac::<1>(trans, b2d)?;
193                if marked.insert(image1) {
194                    // if true, we did not see this dart yet
195                    // i.e. we need to visit it later
196                    min = min.min(image1);
197                    pending.push_back(image1);
198                }
199                let image2 = self.beta_transac::<2>(trans, b0d)?;
200                if marked.insert(image2) {
201                    // if true, we did not see this dart yet
202                    // i.e. we need to visit it later
203                    min = min.min(image2);
204                    pending.push_back(image2);
205                }
206            }
207
208            Ok(min)
209        })
210    }

220    /// Compute the ID of the edge a given dart is part of.
221    ///
222    /// This corresponds to the minimum dart ID among darts composing the 1-cell orbit.
223    ///
224    /// # Errors
225    ///
226    /// This method is meant to be called in a context where the returned `Result` is used to
227    /// validate the transaction passed as argument. Errors should not be processed manually,
228    /// only processed via the `?` operator.
229    pub fn edge_id_transac(
230        &self,
231        trans: &mut Transaction,
232        dart_id: DartIdType,
233    ) -> StmClosureResult<EdgeIdType> {
234        // optimizing this one bc I'm tired
235        let b2 = self.beta_transac::<2>(trans, dart_id)?;
236        if b2 == NULL_DART_ID {
237            Ok(dart_id as EdgeIdType)
238        } else {
239            Ok(b2.min(dart_id) as EdgeIdType)
240        }
241    }

251    /// Compute the ID of the face a given dart is part of.
252    ///
253    /// This corresponds to the minimum dart ID among darts composing the 2-cell orbit.
254    ///
255    /// # Errors
256    ///
257    /// This method is meant to be called in a context where the returned `Result` is used to
258    /// validate the transaction passed as argument. Errors should not be processed manually,
259    /// only processed via the `?` operator.
260    pub fn face_id_transac(
261        &self,
262        trans: &mut Transaction,
263        dart_id: DartIdType,
264    ) -> StmClosureResult<FaceIdType> {
265        AUXILIARIES.with(|t| {
266            let (pending, marked) = &mut *t.borrow_mut();
267            // clear from previous computations
268            pending.clear();
269            marked.clear();
270            // initialize
271            pending.push_back(dart_id);
272            marked.insert(NULL_DART_ID); // we don't want to include the null dart in the orbit
273            marked.insert(dart_id); // we're starting here, so we mark it beforehand
274
275            let mut min = dart_id;
276
277            while let Some(d) = pending.pop_front() {
278                // THIS CODE IS ONLY VALID IN 2D
279                let image1 = self.beta_transac::<1>(trans, d)?;
280                if marked.insert(image1) {
281                    // if true, we did not see this dart yet
282                    // i.e. we need to visit it later
283                    min = min.min(image1);
284                    pending.push_back(image1);
285                }
286                let image2 = self.beta_transac::<0>(trans, d)?;
287                if marked.insert(image2) {
288                    // if true, we did not see this dart yet
289                    // i.e. we need to visit it later

292                }
293            }

299    /// Return an iterator over IDs of all the map's vertices.
300    #[must_use = "unused return value"]
301    pub fn iter_vertices(&self) -> impl Iterator<Item = VertexIdType> + '_ {
302        (1..self.n_darts() as DartIdType)
303            .zip(self.unused_darts.iter().skip(1))
304            .filter_map(
305                |(d, unused)| {
306                    if unused.read_atomic() { None } else { Some(d) }
307                },
308            )
309            .filter_map(|d| {
310                let vid = self.vertex_id(d);
311                if d == vid { Some(vid) } else { None }
312            })
313    }
