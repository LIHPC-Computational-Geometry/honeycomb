7/// **1-(un)sews internals**

9    /// 1-sew transactional implementation.
10    pub(super) fn one_sew(
11        &self,
12        t: &mut Transaction,
13        ld: DartIdType,
14        rd: DartIdType,
15    ) -> TransactionClosureResult<(), SewError> {
16        let b2ld = self.beta_tx::<2>(t, ld)?;
17        let b2l_vid = self.vertex_id_tx(t, b2ld)?;
18        let r_vid = self.vertex_id_tx(t, rd)?;
19
20        try_or_coerce!(self.link::<1>(t, ld, rd), SewError);
21
22        if b2ld != NULL_DART_ID {
23            try_or_coerce!(
24                self.vertices.merge(t, b2l_vid.min(r_vid), b2l_vid, r_vid,),
25                SewError
26            );
27            try_or_coerce!(
28                self.attributes.merge_attributes(
29                    t,
30                    OrbitPolicy::Vertex,
31                    b2l_vid.min(r_vid),
32                    b2l_vid,
33                    r_vid,
34                ),
35                SewError
36            );
37        }
38        Ok(())
39    }
40
41    /// 1-unsew transactional implementation.
42    pub(super) fn one_unsew(
43        &self,
44        t: &mut Transaction,
45        ld: DartIdType,
46    ) -> TransactionClosureResult<(), SewError> {
47        let rd = self.beta_tx::<1>(t, ld)?;
48        let b2ld = self.beta_tx::<2>(t, ld)?;
49
50        try_or_coerce!(self.unlink::<1>(t, ld), SewError);
51
52        if b2ld != NULL_DART_ID {
53            // split vertex attributes
54            let (new_lhs, new_rhs) = (self.vertex_id_tx(t, b2ld)?, self.vertex_id_tx(t, rd)?);
55            try_or_coerce!(
56                self.vertices
57                    .split(t, new_lhs, new_rhs, new_lhs.min(new_rhs)),
58                SewError
59            );
60            try_or_coerce!(
61                self.attributes.split_attributes(
62                    t,
63                    OrbitPolicy::Vertex,
64                    new_lhs,
65                    new_rhs,
66                    new_lhs.min(new_rhs),
67                ),
68                SewError
69            );
70        }
71        Ok(())
72    }
